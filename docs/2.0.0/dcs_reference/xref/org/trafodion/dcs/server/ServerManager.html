92      class RegisteredWatcher implements Watcher {
93          CountDownLatch startSignal;
94  
95          public RegisteredWatcher(CountDownLatch startSignal) {
96              this.startSignal = startSignal;
97          }
98  
99          public void process(WatchedEvent event) {
100             if (event.getType() == Event.EventType.NodeDeleted) {
101                 String znodePath = event.getPath();
102                 LOG.debug("Registered znode deleted [" + znodePath + "]");
103                 try {
104                     startSignal.countDown();
105                 } catch (Exception e) {
106                     e.printStackTrace();
107                     LOG.error(e);
108                 }
109             }
110         }
111     }

113     class ServerMonitor {
114         ScriptContext scriptContext = new ScriptContext();
115         int childInstance;
116         String registeredPath;
117         Stat stat = null;
118         boolean isRunning = false;
119 
120         public ServerMonitor(int childInstance, String registeredPath) {
121             this.childInstance = childInstance;
122             this.registeredPath = registeredPath;
123         }
124 
125         public boolean monitor() throws Exception {
126             LOG.debug("registered path [" + registeredPath + "]");
127             stat = zkc.exists(registeredPath, false);
128             if (stat != null) { // User program znode found in
129                                 // /registered...check pid
130                 isRunning = isPid();
131                 LOG.debug("isRunning [" + isRunning + "]");
132             }
133 
134             return isRunning;
135         }
136 
137         private boolean isPid() throws Exception {
138             String data = Bytes.toString(zkc.getData(registeredPath, false,
139                     stat));
140             Scanner scn = new Scanner(data);
141             scn.useDelimiter(":");
142             scn.next();// state
143             scn.next();// timestamp
144             scn.next();// dialogue Id
145             scn.next();// nid
146             String pid = scn.next();// pid
147             scn.close();
148             scriptContext.setHostName(hostName);
149             scriptContext.setScriptName(Constants.SYS_SHELL_SCRIPT_NAME);
150             scriptContext.setCommand("ps -p " + pid);
151             ScriptManager.getInstance().runScript(scriptContext);
152             return scriptContext.getExitCode() != 0 ? false : true;
153         }
154     }

161         public ServerRunner(int childInstance, String registeredPath) {
162             this.scriptContext = new ScriptContext();
163             this.childInstance = childInstance;
164             this.registeredPath = registeredPath;
165             scriptContext.setHostName(hostName);
166             scriptContext.setScriptName(Constants.SYS_SHELL_SCRIPT_NAME);
167             String command = userProgramCommand
168                     .replace("-ZKHOST", "-ZKHOST " + zkc.getZkQuorum() + " ")
169                     .replace(
170                             "-RZ",
171                             "-RZ " + hostName + ":" + instance + ":"
172                                     + childInstance + " ")
173                     .replace("-ZKPNODE",
174                             "-ZKPNODE " + "\"" + parentZnode + "\"" + " ")
175                     .replace("-CNGTO", "-CNGTO " + connectingTimeout + " ")
176                     .replace("-ZKSTO", "-ZKSTO " + zkSessionTimeout + " ")
177                     .replace("-EADSCO",
178                             "-EADSCO " + userProgExitAfterDisconnect + " ")
179                     .replace("-TCPADD",
180                             "-TCPADD " + netConf.getExtHostAddress() + " ")
181                     .replace("-MAXHEAPPCT",
182                             "-MAXHEAPPCT " + maxHeapPctExit + " ")
183                     .replace(
184                             "-STATISTICSINTERVAL",
185                             "-STATISTICSINTERVAL " + statisticsIntervalTime
186                                     + " ")
187                     .replace("-STATISTICSLIMIT",
188                             "-STATISTICSLIMIT " + statisticsLimitTime + " ")
189                     .replace("-STATISTICSTYPE",
190                             "-STATISTICSTYPE " + statisticsType + " ")
191                     .replace("-STATISTICSENABLE",
192                             "-STATISTICSENABLE " + statisticsEnable + " ")
193                     .replace("-SQLPLAN", "-SQLPLAN " + sqlplanEnable + " ")
194                     .replace("-PORTMAPTOSECS",
195                             "-PORTMAPTOSECS " + userProgPortMapToSecs + " ")
196                     .replace("-PORTBINDTOSECS",
197                             "-PORTBINDTOSECS " + userProgPortBindToSecs)
198                     .replace("&lt;", "<").replace("&amp;", "&")
199                     .replace("&gt;", ">");
200             scriptContext.setCommand(command);
201         }

203         public void exec() throws Exception {
204             cleanupZk();
205             RetryCounter retryCounter = retryCounterFactory.create();
206             while (!isTrafodionRunning()) {
207                 LOG.error("Trafodion is not running");
208                 if (!retryCounter.shouldRetry()) {
209                     return;
210                 } else {
211                     retryCounter.sleepUntilNextRetry();
212                     retryCounter.useRetry();
213                 }
214             }
215             LOG.info("User program exec [" + scriptContext.getCommand() + "]");
216             ScriptManager.getInstance().runScript(scriptContext);// This will
217                                                                  // block while
218                                                                  // user prog is
219                                                                  // running
220             LOG.info("User program exit [" + scriptContext.getExitCode() + "]");
221             StringBuilder sb = new StringBuilder();
222             sb.append("exit code [" + scriptContext.getExitCode() + "]");
223             if (!scriptContext.getStdOut().toString().isEmpty())
224                 sb.append(", stdout [" + scriptContext.getStdOut().toString()
225                         + "]");
226             if (!scriptContext.getStdErr().toString().isEmpty())
227                 sb.append(", stderr [" + scriptContext.getStdErr().toString()
228                         + "]");
229             LOG.info(sb.toString());
230 
231             switch (scriptContext.getExitCode()) {
232             case 3:
233                 LOG.error("Trafodion is not running");
234                 break;
235             case 127:
236                 LOG.error("Cannot find user program executable");
237                 break;
238             default:
239             }
240         }

242         private void cleanupZk() {
243             try {
244                 Stat stat = zkc.exists(registeredPath, false);
245                 if (stat != null)
246                     zkc.delete(registeredPath, -1);
247             } catch (Exception e) {
248                 e.printStackTrace();
249                 LOG.debug(e);
250             }
251         }

253         private boolean isTrafodionRunning() {
254 
255         // Check if Trafodion is up and running
256         // If Trafodion is fully or partially up and operational 
257         // return true else return false.
258         // Invoke sqcheck to check Trafodion status. 
259         // 
260         //   sqcheck returns:
261         //   -1 - Not up ($?=255)
262         //    0 - Fully up and operational
263         //    1 - Partially up and operational
264         //    2 - Partially up and NOT operational
265 
266             ScriptContext scriptContext = new ScriptContext();
267             scriptContext.setHostName(hostName);
268             scriptContext.setScriptName(Constants.SYS_SHELL_SCRIPT_NAME);
269             String command = ("sqcheck");
270             scriptContext.setCommand(command);
271             ScriptManager.getInstance().runScript(scriptContext);// This will
272                                                                  // block while
273                                                                  // script is
274                                                                  // running
275             int exitCode = scriptContext.getExitCode();
276             return (exitCode == 0 || exitCode == 1) ? true : false;
277         }

280     class ServerHandler implements Callable<Integer> {
281         ServerMonitor serverMonitor;
282         ServerRunner serverRunner;
283         int childInstance;
284         String registeredPath;
285         CountDownLatch startSignal = new CountDownLatch(1);
286 
287         public ServerHandler(int childInstance) {
288             this.childInstance = childInstance;
289             this.registeredPath = parentZnode
290                     + Constants.DEFAULT_ZOOKEEPER_ZNODE_SERVERS_REGISTERED
291                     + "/" + hostName + ":" + instance + ":" + childInstance;
292             serverMonitor = new ServerMonitor(childInstance, registeredPath);
293             serverRunner = new ServerRunner(childInstance, registeredPath);
294         }
295 
296         @Override
297         public Integer call() throws Exception {
298             Integer result = new Integer(childInstance);
299 
300             if (serverMonitor.monitor()) {
301                 LOG.info("Server handler [" + instance + ":" + childInstance
302                         + "] is running");
303                 zkc.exists(registeredPath, new RegisteredWatcher(startSignal));
304                 LOG.debug("Waiting for start signal");
305                 startSignal.await();
306                 serverRunner.exec();
307             } else {
308                 LOG.info("Server handler [" + instance + ":" + childInstance
309                         + "] is not running");
310                 serverRunner.exec();
311             }
312 
313             return result;
314         }
315     }

372     @Override
373     public Boolean call() throws Exception {
374 
375         ExecutorService executorService = Executors
376                 .newFixedThreadPool(childServers);
377         CompletionService<Integer> completionService = new ExecutorCompletionService<Integer>(
378                 executorService);
379 
380         try {
381             getMaster();
382             featureCheck();
383             registerInRunning(instance);
384 
385             // When started from bin/dcs-start.sh script childServers will
386             // contain the
387             // count passed in from the servers.sh script. However when
388             // DcsServer is
389             // killed or dies for any reason DcsMaster restarts it using
390             // /bin/dcs-daemon script
391             // which DOES NOT set childServers count.
392             for (int childInstance = 1; childInstance <= childServers; childInstance++) {
393                 completionService.submit(new ServerHandler(childInstance));
394                 LOG.debug("Started server handler [" + instance + ":"
395                         + childInstance + "]");
396             }
397 
398             while (true) {
399                 LOG.debug("Waiting for any server handler to finish");
400                 Future<Integer> f = completionService.take();// blocks waiting
401                                                              // for any
402                                                              // ServerHandler to
403                                                              // finish
404                 if (f != null) {
405                     Integer result = f.get();
406                     LOG.debug("Server handler [" + instance + ":" + result
407                             + "] finished, restarting");
408                     completionService.submit(new ServerHandler(result));
409                 }
410             }
411 
412         } catch (Exception e) {
413             e.printStackTrace();
414             LOG.error(e);
415             if (executorService != null)
416                 executorService.shutdown();
417             throw e;
418         }

458     private void featureCheck() {
459         final String msg1 = "Property "
460                 + Constants.DCS_SERVER_USER_PROGRAM
461                 + " is false. "
462                 + "Please add to your dcs-site.xml file and set <value>false</value> to <value>true</value>.";
463         final String msg2 = "Environment variable $MY_SQROOT is not set.";
464 
465         boolean ready = false;
466         while (!ready) {
467             userProgEnabled = conf.getBoolean(
468                     Constants.DCS_SERVER_USER_PROGRAM,
469                     Constants.DEFAULT_DCS_SERVER_USER_PROGRAM);
470             userProgramHome = System.getProperty("dcs.user.program.home");
471             userProgramCommand = conf.get(
472                     Constants.DCS_SERVER_USER_PROGRAM_COMMAND,
473                     Constants.DEFAULT_DCS_SERVER_USER_PROGRAM_COMMAND);
474 
475             if (userProgEnabled == true && userProgramHome.isEmpty() == false
476                     && userProgramCommand.isEmpty() == false) {
477                 ready = true;
478                 continue;
479             }
480 
481             if (userProgEnabled == false)
482                 LOG.error(msg1);
483             if (userProgramHome.isEmpty())
484                 LOG.error(msg2);
485 
486             try {
487                 Thread.sleep(60000);
488             } catch (InterruptedException e) {
489             }
490         }
491 
492         LOG.info("User program enabled");
493     }

495     private void getMaster() {
496         boolean found = false;
497 
498         while (!found) {
499             LOG.info("Checking DcsMaster znode [" + parentZnode
500                     + Constants.DEFAULT_ZOOKEEPER_ZNODE_MASTER + "]");
501             try {
502                 Stat stat = zkc.exists(parentZnode
503                         + Constants.DEFAULT_ZOOKEEPER_ZNODE_MASTER, false);
504                 if (stat != null) {
505                     List<String> nodes = zkc.getChildren(parentZnode
506                             + Constants.DEFAULT_ZOOKEEPER_ZNODE_MASTER, null);
507                     if (!nodes.isEmpty()) {
508                         StringTokenizer st = new StringTokenizer(nodes.get(0),
509                                 ":");
510                         while (st.hasMoreTokens()) {
511                             masterHostName = st.nextToken();
512                             port = Integer.parseInt(st.nextToken());
513                             portRange = Integer.parseInt(st.nextToken());
514                             masterStartTime = Long.parseLong(st.nextToken());
515                         }
516                         found = true;
517                         LOG.info("DcsMaster znode [" + parentZnode
518                                 + Constants.DEFAULT_ZOOKEEPER_ZNODE_MASTER
519                                 + "] found");
520                     }
521                 } else {
522                     LOG.info("DcsMaster znode [" + parentZnode
523                             + Constants.DEFAULT_ZOOKEEPER_ZNODE_MASTER
524                             + "] not found");
525                 }
526 
527                 if (!found) {
528                     try {
529                         Thread.sleep(5000);
530                     } catch (InterruptedException e) {
531                     }
532                 }
533 
534             } catch (Exception e) {
535                 e.printStackTrace();
536                 LOG.error(e);
537             }
538         }
539     }

541     private void registerInRunning(int instance) {
542         String znode = parentZnode
543                 + Constants.DEFAULT_ZOOKEEPER_ZNODE_SERVERS_RUNNING + "/"
544                 + hostName + ":" + instance + ":" + infoPort + ":"
545                 + System.currentTimeMillis();
546         try {
547             Stat stat = zkc.exists(znode, false);
548             if (stat == null) {
549                 zkc.create(znode, new byte[0], ZooDefs.Ids.OPEN_ACL_UNSAFE,
550                         CreateMode.EPHEMERAL);
551                 LOG.info("Created znode [" + znode + "]");
552             }
553         } catch (KeeperException.NodeExistsException e) {
554             // do nothing...leftover from previous shutdown
555         } catch (Exception e) {
556             e.printStackTrace();
557             LOG.error(e);
558         }
