96      boolean buildConnectReply  (Header hdr, ConnectionContext cc, SocketAddress clientSocketAddress ) throws java.io.UnsupportedEncodingException {
97      
98          boolean replyException = false;
99          Integer serverInstance=0;
100         String serverTimestamp="";
101     
102         versionList = new VersionList();
103     
104         versionList.list[0].componentId = ListenerConstants.DCS_MASTER_COMPONENT;
105         versionList.list[0].majorVersion = ListenerConstants.DCS_MASTER_VERSION_MAJOR_1;
106         versionList.list[0].minorVersion = ListenerConstants.DCS_MASTER_VERSION_MINOR_0;
107         versionList.list[0].buildId	= ListenerConstants.DCS_MASTER_BUILD_1 | ListenerConstants.CHARSET | ListenerConstants.PASSWORD_SECURITY;
108     
109         versionList.list[1].componentId = ListenerConstants.MXOSRVR_ENDIAN + ListenerConstants.ODBC_SRVR_COMPONENT;
110         versionList.list[1].majorVersion = ListenerConstants.MXOSRVR_VERSION_MAJOR;
111         versionList.list[1].minorVersion = ListenerConstants.MXOSRVR_VERSION_MINOR;
112         versionList.list[1].buildId	= ListenerConstants.MXOSRVR_VERSION_BUILD;
113     
114         exception.exception_nr=0;
115         exception.exception_detail=0;
116         exception.ErrorText=null;
117     
118         Stat stat = null;
119         byte[] data = null;
120         boolean found = false;
121         boolean exceptionThrown = false;
122         
123         int length = 0;
124         int index = 0;
125         int maxIndex = -1;
126         int randomPicks = 0;
127         String server = "";
128         
129         try {
130             String registeredPath = parentZnode + Constants.DEFAULT_ZOOKEEPER_ZNODE_SERVERS_REGISTERED;
131             String nodeRegisteredPath = "";
132             
133             if (false == registeredPath.startsWith("/"))
134                 registeredPath = "/" + registeredPath;
135 
136             zkc.sync(registeredPath,null,null);
137             List<String> servers = zkc.getChildren(registeredPath, null);
138             length = servers.size();
139             if (length == 0){
140                 throw new IOException("No Available Servers - length is 0");
141             }
142             if(LOG.isDebugEnabled()){
143                 int dbgLength = length > 4 ? 4 : length;
144                 for (int i = 0; i < dbgLength; i++){
145                     LOG.debug(clientSocketAddress + ": " + i + " server " + servers.get(i) );
146                 }
147             }
148             switch(length) {
149                     case 1: randomPicks = 1;break;
150                     case 2: randomPicks = 1;break;
151                     case 3: case 4: randomPicks = 2;break;
152                     default:randomPicks = 3;
153             }
154             if(LOG.isDebugEnabled())
155                 LOG.debug(clientSocketAddress + ": " + "randomPicks " + randomPicks + ", length " + length );
156             
157             int[] indexArr = new int[length];
158             Arrays.fill(indexArr, 0, length - 1, -1);
159             indexArr[0] = -1;
160 //
161 // pick randomly the AVAILABLE server 
162 //                
163             for(int i=0; i < randomPicks; i++){
164                 while(true){
165                     index = random.nextInt();
166                     index = Math.abs(index);
167                     index %= length;
168                     if (indexArr[index] != index) break;
169                 }
170                 indexArr[index] = index;
171                 maxIndex = Math.max(maxIndex, index);
172                 server = servers.get(index);
173                 nodeRegisteredPath = registeredPath + "/" + server;
174  		if(LOG.isDebugEnabled())
175                     LOG.debug(clientSocketAddress + ": " + " index " + index + " server picked " + server );
176                 data = isServerAvalible(nodeRegisteredPath);
177                 if(data != null){
178                 	found = true;
179                 	break;
180                 }else continue;
181             }
182 //
183 // search sequentially for AVAILABLE server starting from highest random index + 1 to length
184 //
185             if (found == false){
186                 for(index=maxIndex+1; index<length; index++){
187                     if (indexArr[index] != index){
188                         server = servers.get(index);
189                         nodeRegisteredPath = registeredPath + "/" + server;
190                         if(LOG.isDebugEnabled())
191                             LOG.debug(clientSocketAddress + ": " + "server selected in search 1 " + server );
192                             
193                         data = isServerAvalible(nodeRegisteredPath);
194                         if(data != null){
195                         	found = true;
196                         	break;
197                         }else continue;
198                     }
199                 }
200             }
201 //
202 // search sequentially for AVAILABLE server starting from index 0 to max random index - 1
203 //
204             if (found == false){
205                 for(index=0; index<maxIndex; index++){
206                     if (indexArr[index] != index){
207                         server = servers.get(index);
208                         nodeRegisteredPath = registeredPath + "/" + server;
209                         if(LOG.isDebugEnabled())
210                             LOG.debug(clientSocketAddress + ": " + "server selected in search 2 " + server );
211                             
212                         data = isServerAvalible(nodeRegisteredPath);
213                         if(data != null){
214                         	found = true;
215                         	break;
216                         }else continue;
217                     }
218                 }
219             }
220             if (found == true){
221 
222                 String[] stNode = server.split(":");
223                 serverHostName=stNode[0];
224                 serverInstance=Integer.parseInt(stNode[1]);
225                 
226                 String[] stData = (new String(data)).split(":");
227                 timestamp=Long.parseLong(stData[1]);
228                 serverNodeId=Integer.parseInt(stData[3]);
229                 serverProcessId=Integer.parseInt(stData[4]);
230                 serverProcessName=stData[5];
231                 serverIpAddress=stData[6];
232                 serverPort=Integer.parseInt(stData[7]);
233                 
234                 if(LOG.isDebugEnabled()){
235                     LOG.debug(clientSocketAddress + ": " + "serverHostName " + serverHostName );
236                     LOG.debug(clientSocketAddress + ": " + "serverInstance " + serverInstance );
237                     LOG.debug(clientSocketAddress + ": " + "serverNodeId " + serverNodeId );
238                     LOG.debug(clientSocketAddress + ": " + "serverProcessId " + serverProcessId);
239                     LOG.debug(clientSocketAddress + ": " + "serverProcessName " + serverProcessName );
240                     LOG.debug(clientSocketAddress + ": " + "serverIpAddress " + serverIpAddress );
241                     LOG.debug(clientSocketAddress + ": " + "serverPort " + serverPort );
242                     LOG.debug(clientSocketAddress + ": " + "timestamp " + timestamp );
243                 }
244                 
245                 dialogueId = random.nextInt();
246                 dialogueId = (dialogueId < 0 )? -dialogueId : dialogueId;
247                 if(LOG.isDebugEnabled())
248                     LOG.debug(clientSocketAddress + ": " + "dialogueId: " + dialogueId);
249                 data = Bytes.toBytes(String.format("CONNECTING:%d:%d:%d:%d:%s:%s:%d:%s:%s:%s:",
250                         timestamp,
251                         dialogueId, 
252                         serverNodeId,
253                         serverProcessId,
254                         serverProcessName,
255                         serverIpAddress,
256                         serverPort,
257                         cc.computerName, 
258                         clientSocketAddress, 
259                         cc.windowText ));
260                 zkc.setData(nodeRegisteredPath, data, -1);
261             }
262         } catch (KeeperException.NodeExistsException e) {
263             LOG.error(clientSocketAddress + ": " + "do nothing...some other server has created znodes: " + e.getMessage());
264             exceptionThrown = true;
265         } catch (KeeperException e) {
266             LOG.error(clientSocketAddress + ": " + "KeeperException: " + e.getMessage());
267             exceptionThrown = true;
268         } catch (InterruptedException e) {
269             LOG.error(clientSocketAddress + ": " + "InterruptedException: " + e.getMessage());
270             exceptionThrown = true;
271         } catch (IOException ie){
272             LOG.error(clientSocketAddress + ": " + ie.getMessage());
273             exceptionThrown = true;
274         }
275         if (found == false || exceptionThrown == true){
276             exception.exception_nr = ListenerConstants.DcsMasterNoSrvrHdl_exn; //no available servers
277             replyException = true;
278             if(LOG.isDebugEnabled()){
279                 if (found == false)
280                     LOG.info(clientSocketAddress + ": " + "No Available Servers");
281                 else
282                     LOG.info(clientSocketAddress + ": " + "No Available Servers - exception thrown");
283             }
284         } else {
285             
286             if (cc.datasource.length() == 0)
287                 dataSource = "TDM_Default_DataSource";
288             else
289                 dataSource = cc.datasource;
290 
291             if(LOG.isDebugEnabled()){
292                 LOG.debug(clientSocketAddress + ": " + "userName: " + cc.user.userName);
293                 LOG.debug(clientSocketAddress + ": " + "password: XXXXXX");
294                 LOG.debug(clientSocketAddress + ": " + "client: " + cc.client);
295                 LOG.debug(clientSocketAddress + ": " + "location: " + cc.location);
296                 LOG.debug(clientSocketAddress + ": " + "windowText: " + cc.windowText);
297                 LOG.debug(clientSocketAddress + ": " + "dataSource: " + dataSource);
298                 LOG.debug(clientSocketAddress + ": " + "client computer name:ipaddress:port " + cc.computerName+ ":" + clientSocketAddress);
299             }
300 
301             userSid = new String(cc.user.userName).getBytes("UTF-8");
302 
303             isoMapping = 0;
304             StringTokenizer st = new StringTokenizer(serverHostName, ".");
305             if(st.hasMoreTokens()) {
306                 clusterName = String.format("%s_%s", st.nextToken(), cc.client);
307             }
308             else {
309                 clusterName = String.format("%s_%s", serverHostName, cc.client);
310             }
311             if(LOG.isDebugEnabled())
312                 LOG.debug(clientSocketAddress + ": " + "clusterName " + clusterName );
313         }
314     return replyException;
315     }

317 	private byte[] isServerAvalible(String serverPath) throws KeeperException, InterruptedException {
318         Stat stat = zkc.exists(serverPath,false);
319         if(stat != null){
320             byte[] data = zkc.getData(serverPath, false, stat);
321             if(data != null && new String(data).startsWith("AVAILABLE:")){
322             	return data;
323             }
324         }
325 		return null;
326 	}
