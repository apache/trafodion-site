171     ClientData processApi(ClientData clientData) {
172         init();
173         this.clientData = clientData;
174 //
175 // ==============process input ByteBuffer===========================
176 //
177         ByteBuffer bbHeader = clientData.bbHeader;
178         ByteBuffer bbBody = clientData.bbBody;
179         Header hdr = clientData.hdr;
180 
181         bbHeader.flip();
182         bbBody.flip();
183 
184         try {
185             hdr.extractFromByteArray(bbHeader);
186 
187             dialogueId = bbBody.getInt();
188             holdableCursor = bbBody.getInt();
189             queryTimeout = bbBody.getInt();
190             inpRowCnt = bbBody.getInt();
191             maxRowsetSize = bbBody.getInt();
192             sqlStmtType = bbBody.getInt();
193             stmtHandle = bbBody.getInt();
194             stmtType = bbBody.getInt();
195             sqlString = ByteBufferUtils.extractStringWithCharset(bbBody);
196             cursorName = ByteBufferUtils.extractStringWithCharset(bbBody);
197             stmtLabel = ByteBufferUtils.extractStringWithCharset(bbBody);
198             stmtExplainLabel = ByteBufferUtils.extractString(bbBody);
199 //            txId = ByteBufferUtils.extractByteArray(bbBody);
200             if(LOG.isDebugEnabled()) {
201                 LOG.debug(serverWorkerName + ". dialogueId :" + dialogueId);
202                 LOG.debug(serverWorkerName + ". holdableCursor :" + holdableCursor);
203                 LOG.debug(serverWorkerName + ". queryTimeout :" + queryTimeout);
204                 LOG.debug(serverWorkerName + ". inpRowCnt :" + inpRowCnt);
205                 LOG.debug(serverWorkerName + ". maxRowsetSize :" + maxRowsetSize);
206                 LOG.debug(serverWorkerName + ". sqlStmtType :" + sqlStmtType);
207                 LOG.debug(serverWorkerName + ". stmtHandle :" + stmtHandle);
208                 LOG.debug(serverWorkerName + ". stmtType :" + stmtType);
209                 LOG.debug(serverWorkerName + ". sqlString :" + sqlString);
210                 LOG.debug(serverWorkerName + ". cursorName :" + cursorName);
211                 LOG.debug(serverWorkerName + ". stmtLabel :" + stmtLabel);
212                 LOG.debug(serverWorkerName + ". stmtExplainLabel :" + stmtExplainLabel);
213 //                LOG.debug(serverWorkerName + ". txId :" + txId);
214             }
215             if (dialogueId < 1 ) {
216                 throw new SQLException(serverWorkerName + ". Wrong dialogueId :" + dialogueId);
217             }
218             if (dialogueId != clientData.getDialogueId() ) {
219                 throw new SQLException(serverWorkerName + ". Wrong dialogueId sent by the Client [sent/expected] : [" + dialogueId + "/" + clientData.getDialogueId() + "]");
220             }
221 //=============================================================================
222             trafConn = clientData.getTrafConnection();
223             trafStmt = trafConn.getTrafStatement(stmtLabel, stmtHandle);
224 
225             isResultSet = trafStmt.getIsResultSet();
226             isSpj = trafStmt.getIsSpj();
227             stmt = trafStmt.getStatement();
228             if (isSpj == true) cstmt = (CallableStatement)stmt; else pstmt = (PreparedStatement)stmt;
229 //
230 //=====================Process ServerApiSqlExecute===========================
231 //
232             try {
233                 int addedBatches = 0;
234                 paramCount = trafStmt.getParamCount();
235                 if (paramCount > 0) {
236                     long startOffset;
237                     int rowCount;
238                     int paramRowCount;
239                     int paramMode;
240 
241                     if(LOG.isDebugEnabled())
242                     ByteBufferUtils.toHexString("inValues", bbBody, 10);
243                     paramDescList = trafStmt.getParamDescList();
244                     paramLength = trafStmt.getParamLength();
245                     paramValuesLength = bbBody.getInt();
246                     if(LOG.isDebugEnabled()) {
247                         LOG.debug("paramValuesLength :" + paramValuesLength);
248                         LOG.debug("paramLength :" + paramLength);
249                     }
250                     if (paramValuesLength > 0) {
251                         startOffset = bbBody.position();
252                         rowCount = inpRowCnt == 0 ? 1 : inpRowCnt;
253                         paramRowCount = rowCount;
254                         paramMode = 0;
255                         for (int row = 0; row < rowCount; row++) {
256                             for (int param = 1; param <= paramCount; param++) {
257                                 paramDesc = paramDescList.getDescriptors2()[param - 1];
258                                 paramMode = paramDesc.getParamMode();
259                                 if ( paramMode == java.sql.ParameterMetaData.parameterModeOut) continue;
260                                 bbBody = setData(paramDesc, paramRowCount, param, row, bbBody, startOffset);
261                             }
262                             if (rowCount > 1) {
263                                 pstmt.addBatch(); // no SPJ
264                                 addedBatches++;
265                             }
266                             if(LOG.isDebugEnabled())
267                             LOG.debug("endof row :" + row + " addedBatches :" + addedBatches);
268                         }
269                     }
270                     if(LOG.isDebugEnabled())
271                         LOG.debug("addedBatches :" + addedBatches);
272 
273                     for (int param = 1; param <= paramCount; param++) {
274                         paramDesc = paramDescList.getDescriptors2()[param - 1];
275                         paramMode = paramDesc.getParamMode();
276                         if ( paramMode == java.sql.ParameterMetaData.parameterModeIn) continue;
277                         if(LOG.isDebugEnabled())
278                         LOG.debug("registerOutParameter :" + param);
279                         regParamCount++;
280                         cstmt.registerOutParameter(param, java.sql.Types.VARCHAR);
281                     }
282                 }
283                 int txIdLen = bbBody.limit() - bbBody.position();
284                 if (txIdLen > 0) {
285                     txId = new byte[txIdLen];
286                     bbBody.get(txId);
287                 }
288                 if(LOG.isDebugEnabled())
289                     LOG.debug(serverWorkerName + ". txId :" + Arrays.toString(txId));
290 
291                 if (addedBatches > 1) {
292 
293                     if(LOG.isDebugEnabled())
294                         LOG.debug(serverWorkerName + ". T2 Execute.executeBatch()");
295 
296                     int[] numStatus = pstmt.executeBatch();
297                     rowsAffected = numStatus.length;
298 
299                     if(LOG.isDebugEnabled())
300                         LOG.debug(serverWorkerName + ". T2 Execute.executeBatch rowsAffected :" + rowsAffected);
301                 }
302                 else if (isResultSet) {
303                     if(LOG.isDebugEnabled())
304                         LOG.debug(serverWorkerName + ". T2 Execute.executeQuery()");
305                     rs = pstmt.executeQuery();
306                     if (rs != null) {
307                         outDescList = trafStmt.getOutDescList();
308                         trafStmt.addTResultSet(new TrafResultSet(rs, 0, stmtLabel, 0, trafStmt.getOutDescList(),""));
309                     }
310                 } else if (isSpj == true) {
311                     if(LOG.isDebugEnabled())
312                         LOG.debug(serverWorkerName + ". SPJ T2 Execute.execute()");
313                     boolean rsAvailable = cstmt.execute();
314                     if (regParamCount > 0) { //out, inout parameters
315                         rowsAffected = 1;
316                         outValuesLength = (int)paramLength;
317                         outValues = new byte[outValuesLength];
318                         outValues = buildOutValues(paramDescList, outValues, paramCount, cstmt, bbBody.order());
319                     }
320                     if(LOG.isDebugEnabled())
321                         LOG.debug(serverWorkerName + ". SPJ T2 Execute.rsAvailable :" + rsAvailable);
322                     if (rsAvailable == true) {
323                         setMultipleResultSets(cstmt);
324                     }
325                 } else {
326                     resultSetCount = 0;
327                     rowsAffected = pstmt.executeUpdate();
328                     if(LOG.isDebugEnabled())
329                         LOG.debug(serverWorkerName + ". T2 Execute.executeUpdate() rowsAffected :" + rowsAffected);
330                 }
331             } catch (BatchUpdateException bex) {
332                 LOG.error(serverWorkerName + ". Execute.BatchUpdateException " + bex);
333                 errorList = new SQLWarningOrErrorList(bex, bex.getUpdateCounts());
334                 returnCode = errorList.getReturnCode();
335 
336             } catch (SQLException ex) {
337                 LOG.error(serverWorkerName + ". Execute.SQLException " + ex);
338                 errorList = new SQLWarningOrErrorList(ex);
339                 returnCode = errorList.getReturnCode();
340             }
341             if (sqlString.isEmpty() == false)
342                 sqlQueryType = SqlUtils.getSqlStmtType(sqlString);
343             else
344                 sqlQueryType = SqlUtils.getSqlStmtType(sqlStmtType);
345 //
346 //===================calculate length of output ByteBuffer========================
347 //
348             if(LOG.isDebugEnabled())
349                 LOG.debug(serverWorkerName + ". calculate length of output ByteBuffer");
350             bbHeader.clear();
351             bbBody.clear();
352 //
353 // check if ByteBuffer is big enough for output
354 //
355             TrafResultSet trs = null;
356 
357             int dataLength = ServerConstants.INT_FIELD_SIZE;//returnCode
358             if (errorList != null)
359                 dataLength += errorList.lengthOfData();
360             else
361                 dataLength += ServerConstants.INT_FIELD_SIZE;//totalErrorLength = 0
362 
363             if (outDescList != null)
364                 dataLength += outDescList.lengthOfData();
365             else
366                 dataLength += ServerConstants.INT_FIELD_SIZE;//outDescLength = 0
367 
368             dataLength += ServerConstants.INT_FIELD_SIZE;//rowsAffected
369             dataLength += ServerConstants.INT_FIELD_SIZE;//sqlQueryType
370             dataLength += ServerConstants.INT_FIELD_SIZE;//estimatedCost
371             dataLength += ByteBufferUtils.lengthOfByteArray(outValues);//outValues
372 
373             dataLength += ServerConstants.INT_FIELD_SIZE;// SPJ Result Sets
374             if (resultSetCount > 0) {
375                 boolean bNextResultSet = true;
376                 trafStmt.setFirstTResultSet();
377                 while (bNextResultSet) {
378                     trs = trafStmt.getTrafResultSet();
379                     dataLength += trs.lengthOfData();
380                     bNextResultSet = trafStmt.getNextTResultSet();
381                 }
382             }
383             dataLength += ByteBufferUtils.lengthOfString(singleSyntax);
384 
385             int availableBuffer = bbBody.capacity() - bbBody.position();
386             if(LOG.isDebugEnabled())
387                 LOG.debug(serverWorkerName + ". dataLength :" + dataLength + " availableBuffer :" + availableBuffer);
388             if (dataLength > availableBuffer )
389                 bbBody = ByteBufferUtils.increaseCapacity(bbBody, dataLength > ServerConstants.BODY_SIZE ? dataLength : ServerConstants.BODY_SIZE );
390 //
391 //===================== build output ==============================================
392 //
393             if(LOG.isDebugEnabled()) {
394                 LOG.debug(serverWorkerName + ". returnCode :" + returnCode);
395                 LOG.debug(serverWorkerName + ". errorList :" + errorList);
396                 LOG.debug(serverWorkerName + ". outDescList :" + outDescList);
397                 LOG.debug(serverWorkerName + ". rowsAffected :" + rowsAffected);
398                 LOG.debug(serverWorkerName + ". sqlQueryType :" + sqlQueryType);
399                 LOG.debug(serverWorkerName + ". estimatedCost :" + estimatedCost);
400                 LOG.debug(serverWorkerName + ". outValues :" + outValues);
401                 LOG.debug(serverWorkerName + ". SPJ Result Sets :" + resultSetCount);
402                 LOG.debug(serverWorkerName + ". singleSyntax :" + singleSyntax);
403             }
404             bbBody.putInt(returnCode);
405             if (errorList != null)
406                 errorList.insertIntoByteBuffer(bbBody);
407             else
408                 bbBody.putInt(0);
409 
410             if (outDescList != null) {
411                 outDescList.insertIntoByteBuffer(bbBody);
412             } else
413                 bbBody.putInt(0);
414 
415             ByteBufferUtils.insertUInt(rowsAffected, bbBody);
416             bbBody.putInt(sqlQueryType);
417             bbBody.putInt(estimatedCost);
418             ByteBufferUtils.insertByteArray(outValues, bbBody); //outValues
419             bbBody.putInt(resultSetCount);
420 
421             if (resultSetCount > 0) {
422                 boolean bNextResultSet = true;
423                 trafStmt.setFirstTResultSet();
424                 while (bNextResultSet) {
425                     trs = trafStmt.getTrafResultSet();
426                     trs.insertIntoByteBuffer(bbBody);
427                     bNextResultSet = trafStmt.getNextTResultSet();
428                 }
429                 trafStmt.setFirstTResultSet();
430             }
431             ByteBufferUtils.insertString(singleSyntax, bbBody);
432 
433             bbBody.flip();
434 //=========================Update header================================
435             hdr.setTotalLength(bbBody.limit());
436             hdr.insertIntoByteBuffer(bbHeader);
437             bbHeader.flip();
438 
439             clientData.setByteBufferArray(bbHeader, bbBody);
440             clientData.setHdr(hdr);
441             clientData.setRequest(ServerConstants.REQUST_WRITE_READ);
442 
443         } catch (UnsupportedEncodingException ue) {
444             LOG.error(serverWorkerName + ". Execute.UnsupportedEncodingException :" + ue);
445             clientData.setRequestAndDisconnect();
446         } catch (Exception e) {
447             LOG.error(serverWorkerName + ". Execute.Exception :" + e);
448             clientData.setRequestAndDisconnect();
449         }
450         reset();
451         return clientData;
452     }

455     ByteBuffer setData(Descriptor2 paramDesc, int paramRowCount, int param, int row, ByteBuffer bbBody, long startOffset) throws SQLException, ParseException, UnsupportedEncodingException {
456         int paramLength = 0;
457 
458         String tmpStr = "";
459         Short tmps = 0;
460         Integer tmpi = 0;
461         Long tmpl = 0L;
462         Float tmpf = 0f;
463         Double tmpd = 0d;
464         BigInteger tmpbi;
465         BigDecimal tmpbd;
466         boolean isSigne = true;
467 
468         Object retObj;
469         byte[] tbuffer = null;
470         int year, month, day, hour, minute, second;
471         long nanoSeconds;
472         String charSet = "";
473 
474         int precision = paramDesc.getPrecision();
475         int scale = paramDesc.getScale();
476         int datetimeCode = paramDesc.getDatetimeCode();
477         int FSDataType = paramDesc.getFsDataType();
478         int OdbcDataType = paramDesc.getOdbcDataType();
479         int dataCharSet = paramDesc.getSqlCharset();
480         int dataLength = paramDesc.getMaxLen();
481         int dataType = paramDesc.getDataType();
482 
483         // setup the offsets
484         int noNullValue = paramDesc.getNoNullValue();
485         int nullValue = paramDesc.getNullValue();
486         long nullOffset = 0L;
487         long noNullOffset = 0L;
488 
489         if (dataType == ServerConstants.SQLTYPECODE_VARCHAR_WITH_LENGTH) {
490             dataLength += 2;
491 
492             if (dataLength % 2 != 0)
493                 dataLength++;
494         }
495         if(LOG.isDebugEnabled())
496             LOG.debug("nullValue :"+ nullValue);
497         if (nullValue != -1) {
498             nullValue = (nullValue * paramRowCount) + (row * 2);
499             nullOffset = startOffset + nullValue;
500             bbBody.position((int)nullOffset);
501             short isNull = bbBody.getShort();
502             if(LOG.isDebugEnabled())
503                 LOG.debug("isNull :"+ isNull);
504             if ( isNull == -1) {
505                 if (isSpj == true) cstmt.setObject(param,null); else pstmt.setObject(param,null);
506                 return bbBody;
507             }
508         }
509         noNullValue = (noNullValue * paramRowCount) + (row * dataLength);
510         noNullOffset = startOffset + noNullValue;
511         bbBody.position((int)noNullOffset);
512 
513         if(LOG.isDebugEnabled())
514             LOG.debug("[" + param + "] noNullOffset :"+ noNullOffset + " noNullValue :" + noNullValue);
515 
516         if(dataCharSet == SqlUtils.SQLCHARSETCODE_UNICODE)
517             charSet = "UTF-16LE";
518         else
519             charSet = SqlUtils.getCharsetName(dataCharSet);
520 
521         switch (dataType) {
522             case ServerConstants.SQLTYPECODE_CHAR:
523                 if(LOG.isDebugEnabled())
524                     LOG.debug("["+ param + "] dataType :SQLTYPECODE_CHAR");
525                 tbuffer = new byte[dataLength];
526                 bbBody.get(tbuffer, 0, dataLength);
527                 retObj = new String(tbuffer, charSet);
528                 if(LOG.isDebugEnabled())
529                     LOG.debug("["+ param + "] dataType :SQLTYPECODE_CHAR length :" + dataLength + " scale :" + scale + "  retObj :" + retObj);
530                 break;
531             case ServerConstants.SQLTYPECODE_VARCHAR:
532             case ServerConstants.SQLTYPECODE_VARCHAR_LONG:
533                 if(LOG.isDebugEnabled())
534                     ByteBufferUtils.toHexString("VARCHAR/VARCHAR_LONG", bbBody, (dataLength > 100 ? 100 : dataLength));
535                 tbuffer = new byte[dataLength];
536                 bbBody.get(tbuffer, 0, dataLength);
537                 retObj = new String(tbuffer, charSet);
538                 if(LOG.isDebugEnabled()) {
539                     LOG.debug("["+ param + "] dataType :SQLTYPECODE_VARCHAR len :" + dataLength + " scale :" + scale + "  retObj :" + retObj);
540                     LOG.debug(" retObj :" + Arrays.toString(((String)retObj).getBytes()));
541                 }
542                 break;
543             case ServerConstants.SQLTYPECODE_VARCHAR_WITH_LENGTH:
544                 if(LOG.isDebugEnabled())
545                     ByteBufferUtils.toHexString("VARCHAR_WITH_LENGTH", bbBody, (dataLength > 100 ? 100 : dataLength));
546                 boolean shortLength = precision < Math.pow(2, 15);
547                 int dataOffset = (shortLength) ? 2 : 4;
548                 dataLength = (shortLength) ? bbBody.getShort() : bbBody.getInt();
549                 tbuffer = new byte[dataLength];
550                 bbBody.get(tbuffer, 0, dataLength);
551                 retObj = new String(tbuffer, charSet);
552                 if(LOG.isDebugEnabled()) {
553                     LOG.debug("["+ param + "] dataType :SQLTYPECODE_VARCHAR len :" + dataLength + " scale :" + scale + "  retObj :" + retObj);
554                     LOG.debug(" retObj :" + Arrays.toString(((String)retObj).getBytes()));
555                 }
556                 break;
557             case ServerConstants.SQLTYPECODE_INTERVAL:
558                 tbuffer = new byte[dataLength];
559                 bbBody.get(tbuffer);
560                 retObj = new String(tbuffer);
561                 if(LOG.isDebugEnabled())
562                     LOG.debug("["+ param + "] dataType :SQLTYPECODE_INTERVAL dataLength :" + dataLength + " scale :" + scale + "  retObj :" + retObj);
563                 break;
564             case ServerConstants.SQLTYPECODE_DATETIME:
565                 switch (datetimeCode) {
566                     case ServerConstants.SQLDTCODE_DATE:
567                         tbuffer = new byte[dataLength];
568                         bbBody.get(tbuffer);
569                         retObj = java.sql.Date.valueOf((new String(tbuffer)).trim());
570                         if(LOG.isDebugEnabled())
571                             LOG.debug("["+ param + "] dataType :SQLDTCODE_DATE dataLength :" + dataLength + " scale :" + scale + "  retObj :" + retObj);
572                         break;
573                     case ServerConstants.SQLDTCODE_TIMESTAMP:
574                         tbuffer = new byte[dataLength];
575                         bbBody.get(tbuffer);
576                         retObj = Timestamp.valueOf((new String(tbuffer)).trim());
577                         if(LOG.isDebugEnabled())
578                             LOG.debug("["+ param + "] dataType :SQLDTCODE_TIMESTAMP dataLength :" + dataLength + " scale :" + scale + "  retObj :" + retObj);
579                         break;
580                     case ServerConstants.SQLDTCODE_TIME:
581                         if (OdbcDataType == java.sql.Types.OTHER) // For Desc.SQLDTCODE_HOUR_TO_FRACTION
582                         {
583                             tbuffer = new byte[dataLength];
584                             bbBody.get(tbuffer);
585                             retObj = new String((new String(tbuffer)).trim());
586                         } else {
587                             dataLength = ServerConstants.timeLength;
588                             tbuffer = new byte[dataLength];
589                             bbBody.get(tbuffer);
590                             retObj = Time.valueOf((new String(tbuffer)).trim());
591                         }
592                         if(LOG.isDebugEnabled())
593                             LOG.debug("["+ param + "] dataType :SQLDTCODE_TIME dataLength :" + dataLength + " scale :" + scale + "  retObj :" + retObj);
594                         break;
595                     default:
596                         tbuffer = new byte[dataLength];
597                         bbBody.get(tbuffer);
598                         retObj = new String(tbuffer);
599                         if(LOG.isDebugEnabled())
600                             LOG.debug("["+ param + "] dataType :default dataLength :" + dataLength + " scale :" + scale + "  retObj :" + retObj);
601                         break;
602                 } // SQLTYPECODE_DATETIME
603                 break;
604             case ServerConstants.SQLTYPECODE_SMALLINT:
605                 short sValue = bbBody.getShort();
606                 retObj = new Short(sValue);
607                 if (scale > 0) {
