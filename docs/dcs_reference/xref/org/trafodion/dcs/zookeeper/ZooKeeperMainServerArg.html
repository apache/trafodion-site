24  package org.trafodion.dcs.zookeeper;
25  
26  import java.util.Properties;
27  import java.util.Map.Entry;
28  
29  import org.apache.hadoop.conf.Configuration;
30  import org.trafodion.dcs.util.DcsConfiguration;
31  
32  /**
33   * Tool for reading a ZooKeeper server from HBase XML configuration producing
34   * the '-server host:port' argument to pass ZooKeeperMain.  This program
35   * emits either '-server HOST:PORT" where HOST is one of the zk ensemble
36   * members plus zk client port OR it emits '' if no zk servers found (Yes,
37   * it emits '-server' too).
38   */
39  public class ZooKeeperMainServerArg {
40    public String parse(final Configuration c) {
41      // Note that we do not simply grab the property
42      // Constants.ZOOKEEPER_QUORUM from the DcsConfiguration because the
43      // user may be using a zoo.cfg file.
44      Properties zkProps = ZKConfig.makeZKProps(c);
45      String host = null;
46      String clientPort = null;
47      for (Entry<Object, Object> entry: zkProps.entrySet()) {
48        String key = entry.getKey().toString().trim();
49        String value = entry.getValue().toString().trim();
50        if (key.startsWith("server.") && host == null) {
51          String[] parts = value.split(":");
52          host = parts[0];
53        } else if (key.endsWith("clientPort")) {
54          clientPort = value;
55        }
56        if (host != null && clientPort != null) break;
57      }
58      return host != null && clientPort != null? host + ":" + clientPort: null;
59    }
60  
61    /**
62     * Run the tool.
63     * @param args Command line arguments. First arg is path to zookeepers file.
64     */
65    public static void main(String args[]) {
66      Configuration conf = DcsConfiguration.create();
67      String hostport = new ZooKeeperMainServerArg().parse(conf);
68      System.out.println((hostport == null || hostport.length() == 0)? "":
69        "-server " + hostport);
70    }
71  }
