158     static public short getSqlStmtType(String str) {
159         //
160         // Kludge to determin if the type of statement.
161         //
162         String tokens[] = str.split("[^a-zA-Z]+", 3);
163         short rt1 = ServerConstants.TYPE_UNKNOWN;
164         String str3 = "";
165         //
166         // If there are no separators (i.e. no spaces, {, =, etc.) in front of
167         // the
168         // first token, then the first token is the key word we are looking for.
169         // Else, the first token is an empty string (i.e. split thinks the first
170         // token is the empty string followed by a separator), and the second
171         // token is the key word we are looking for.
172         //
173         if (tokens[0].length() > 0) {
174             str3 = tokens[0].toUpperCase();
175         } else {
176             str3 = tokens[1].toUpperCase();
177         }
178         if ((str3.equals("SELECT")) || (str3.equals("SHOWSHAPE"))
179                 || (str3.equals("INVOKE")) || (str3.equals("SHOWCONTROL"))
180                 || (str3.equals("SHOWPLAN"))) {
181               rt1 = ServerConstants.SQL_SELECT_NON_UNIQUE;
182 //            rt1 = ServerConstants.TYPE_SELECT;
183        } else if (str3.equals("UPDATE")) {
184               rt1 = ServerConstants.SQL_UPDATE_NON_UNIQUE;
185 //            rt1 = ServerConstants.TYPE_UPDATE;
186         } else if (str3.equals("DELETE")) {
187               rt1 = ServerConstants.SQL_DELETE_NON_UNIQUE;
188 //            rt1 = ServerConstants.TYPE_DELETE;
189         } else if (str3.equals("INSERT") || (str.equals("UPSERT"))) {
190             if (str.indexOf('?') == -1) {
191               rt1 = ServerConstants.SQL_INSERT_NON_UNIQUE;
192 //            rt1 = ServerConstants.TYPE_INSERT;
193             } else {
194                 rt1 = ServerConstants.TYPE_INSERT_PARAM;
195             }
196         } else if (str3.equals("EXPLAIN")) {
197             rt1 = ServerConstants.TYPE_EXPLAIN;
198         } else if (str3.equals("CREATE")) {
199             rt1 = ServerConstants.TYPE_CREATE;
200         } else if (str3.equals("GRANT")) {
201             rt1 = ServerConstants.TYPE_GRANT;
202         } else if (str3.equals("DROP")) {
203             rt1 = ServerConstants.TYPE_DROP;
204         } else if (str3.equals("CALL")) {
205             rt1 = ServerConstants.TYPE_CALL;
206         } else if (str3.equals("EXPLAIN")) {
207             rt1 = ServerConstants.TYPE_EXPLAIN;
208         } else if (str3.equals("CONTROL")) {
209             rt1 = ServerConstants.TYPE_CONTROL;
210         } else {
211             rt1 = ServerConstants.TYPE_UNKNOWN;
212         }
213         return rt1;

489     static public BigDecimal getBigDecimalValue(Object paramValue) throws SQLException {
490         BigDecimal tmpbd;
491 
492         if (paramValue instanceof Long) {
493             tmpbd = BigDecimal.valueOf(((Long) paramValue).longValue());
494         } else if (paramValue instanceof Integer) {
495             tmpbd = BigDecimal.valueOf(((Integer) paramValue).longValue());
496         } else if (paramValue instanceof BigDecimal) {
497             tmpbd = (BigDecimal) paramValue;
498         } else if (paramValue instanceof String) {
499             String sVal = (String) paramValue;
500             if (sVal.equals("true") == true) {
501                 sVal = "1";
502             } else if (sVal.equals("false") == true) {
503                 sVal = "0";
504             }
505             tmpbd = new BigDecimal(sVal);
506         } else if (paramValue instanceof Float) {
507             tmpbd = new BigDecimal(paramValue.toString());
508         } else if (paramValue instanceof Double) {
509             tmpbd = new BigDecimal(((Double) paramValue).toString());
510         } else if (paramValue instanceof Boolean) {
511             tmpbd = BigDecimal.valueOf(((((Boolean) paramValue).booleanValue() == true) ? 1 : 0));
512         } else if (paramValue instanceof Byte) {
513             tmpbd = BigDecimal.valueOf(((Byte) paramValue).longValue());
514         } else if (paramValue instanceof Short) {
515             tmpbd = BigDecimal.valueOf(((Short) paramValue).longValue());
516         } else if (paramValue instanceof Integer) {
517             tmpbd = BigDecimal.valueOf(((Integer) paramValue).longValue());
518             // For LOB Support SB: 10/25/2004
519             /*
520              * else if (paramValue instanceof DataWrapper) tmpbd =
521              * BigDecimal.valueOf(((DataWrapper)paramValue).longValue);
522              */
523         } else {
524             throw new SQLException("object_type_not_supported");
525         }
526         return tmpbd;
527     } // end getBigDecimalValue

529     static public BigDecimal convertSQLBigNumToBigDecimal(ByteBuffer sourceData, int len, int scale, boolean isUnSigned) {
530         String strVal = ""; // our final String
531         boolean negative = false;
532 
533         // we need the data in an array which can hold UNSIGNED 16 bit values
534         // in java we dont have unsigned datatypes so 32-bit signed is the best
535         // we can do
536         int[] dataInShorts = new int[len / 2];
537         for (int i = 0; i < dataInShorts.length; i++){
538             dataInShorts[i] = ByteBufferUtils.extractUShort(sourceData); // copy
539         // the
540         // data
541             LOG.debug("dataInShorts[" + i + "] :" + dataInShorts[i]);
542         }
543         if (isUnSigned == false){
544             if ((dataInShorts[dataInShorts.length - 1] & 0xFF00) > 0){
545                 negative = ((dataInShorts[dataInShorts.length - 1] & 0x8000) > 0);
546                 dataInShorts[dataInShorts.length - 1] &= 0x7FFF; // force sign to 0, continue
547             }
548             else {
549                 negative = ((dataInShorts[dataInShorts.length - 1] & 0x0080) > 0);
550                 dataInShorts[dataInShorts.length - 1] &= 0xFF7F; // force sign to 0, continue
551             }
552             // normally
553         }
554 
555         int curPos = dataInShorts.length - 1; // start at the end
556         while (curPos >= 0 && dataInShorts[curPos] == 0)
557             // get rid of any trailing 0's
558             curPos--;
559 
560         int remainder = 0;
561         long temp; // we need to use a LONG since we will have to hold up to
562         // 32-bit UNSIGNED values
563 
564         // we now have the huge value stored in 2 bytes chunks
565         // we will divide by 10000 many times, converting the remainder to
566         // String
567         // when we are left with a single chunk <10000 we will handle it using a
568         // special case
569         while (curPos >= 0 || dataInShorts[0] >= 10000) {
570             // start on the right, divide the 16 bit value by 10000
571             // use the remainder as the upper 16 bits for the next division
572             for (int j = curPos; j >= 0; j--) {
573                 // these operations got messy when java tried to infer what size
574                 // to store the value in
575                 // leave these as separate operations for now...always casting
576                 // back to a 64 bit value to avoid sign problems
577                 temp = remainder;
578                 temp &= 0xFFFF;
579                 temp = temp << 16;
580                 temp += dataInShorts[j];
581 
582                 dataInShorts[j] = (int) (temp / 10000);
583                 remainder = (int) (temp % 10000);
584             }
585 
586             // if we are done with the current 16bits, move on
587             if (dataInShorts[curPos] == 0)
588                 curPos--;
589
