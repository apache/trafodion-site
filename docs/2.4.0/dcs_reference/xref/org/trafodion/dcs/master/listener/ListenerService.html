156     public void run () {
157         try {
158             selector = SelectorProvider.provider().openSelector();
159             if(LOG.isDebugEnabled())
160                 LOG.debug("ServerSocketChannel.open()");
161             server = ServerSocketChannel.open();
162             server.configureBlocking(false);
163 //          InetSocketAddress isa = new InetSocketAddress(ia, port);
164             InetSocketAddress isa = new InetSocketAddress(port); //use any ip address for this port
165             server.socket().bind(isa);
166             SelectionKey serverkey = server.register(selector, SelectionKey.OP_ACCEPT );
167             int keysAdded = 0;
168             PendingRequest preq = null;
169 
170             while(true){
171                 synchronized (this.pendingChanges) {
172                     Iterator<PendingRequest> changes = this.pendingChanges.iterator();
173                     while (changes.hasNext()) {
174                         preq = changes.next();
175                         SelectionKey key = preq.key;
176                         int request = preq.request;
177                         switch(request){
178                         case ListenerConstants.REQUST_WRITE_EXCEPTION:
179                             if(metrics != null) metrics.listenerNoAvailableServers();
180                         case ListenerConstants.REQUST_WRITE:
181                             key.interestOps(key.interestOps() | SelectionKey.OP_WRITE);
182                             break;
183                         case ListenerConstants.REQUST_CLOSE:
184                             try {
185                                 SocketChannel client = (SocketChannel)key.channel();
186                                 client.close();
187                                 ClientData clientData = (ClientData) key.attachment();
188                                 clientData.header = null;
189                                 clientData.body = null;
190                                 clientData.hdr = null;
191                                 clientData.conectContex = null;
192                                 clientData.clientSocketAddress = null;
193                                 key.cancel();
194                                 if(metrics != null) metrics.listenerRequestRejected();
195                             } catch (IOException e) {}
196                             break;
197                         }
198                         preq.key = null;
199                     }
200                     this.pendingChanges.clear();
201                 }
202 
203                 while ((keysAdded = selector.select(setSelectorTimeout())) > 0) {
204                     Set<SelectionKey> keys = selector.selectedKeys();
205                     Iterator<SelectionKey> i = keys.iterator();
206 
207                     while (i.hasNext()) {
208                         SelectionKey key = i.next();
209                         i.remove();
210                         if (!key.isValid()) continue;
211 
212                         if ((key.readyOps() & SelectionKey.OP_ACCEPT) == SelectionKey.OP_ACCEPT) {
213                             if(LOG.isDebugEnabled())
214                                 LOG.debug("Ready to process ACCEPT");
215                             processAccept(key);
216                         } else if ((key.readyOps() & SelectionKey.OP_READ) == SelectionKey.OP_READ) {
217                             if(LOG.isDebugEnabled())
218                                 LOG.debug("Ready to process READ");
219                             processRead(key);
220                         } else if ((key.readyOps() & SelectionKey.OP_WRITE) == SelectionKey.OP_WRITE) {
221                             if(LOG.isDebugEnabled())
222                                 LOG.debug("Ready to process WRITE");
223                             processWrite(key);
224                         }
225                     }
226                 }
227                 if (false == timeouts.isEmpty()){
228                     long currentTime = System.currentTimeMillis();
229                     Iterator<SelectionKey> i = timeouts.keySet().iterator();
230                     while(i.hasNext()){
231                       SelectionKey key = i.next();
232                       if (currentTime - timeouts.get(key) > requestTimeout){
233                           long timeout = (currentTime - timeouts.get(key))/1000;
234                           SocketChannel client = (SocketChannel) key.channel();
235                           Socket s = client.socket();
236                           if ((key.interestOps() & SelectionKey.OP_READ) == SelectionKey.OP_READ){
237                               if(LOG.isDebugEnabled())
238                                   LOG.debug("Read from client timeouted[" + timeout + " seconds] from: " + s.getRemoteSocketAddress());
239                               if(metrics != null) metrics.listenerReadTimeout();
240                           }
241                           else if ((key.interestOps() & SelectionKey.OP_WRITE) == SelectionKey.OP_WRITE) {
242                               if(LOG.isDebugEnabled())
243                                   LOG.debug("Write to client timeouted[" + timeout + " seconds] from: " + s.getRemoteSocketAddress());
244                               if(metrics != null) metrics.listenerWriteTimeout();
245                           }
246                           else {
247                               if(LOG.isDebugEnabled())
248                                   LOG.debug("Client timeouted[" + timeout + " seconds] from: " + s.getRemoteSocketAddress());
249                               if(metrics != null) metrics.listenerRequestRejected();
250                           }
251                           try {
252                               client.close();
253                           } catch (IOException ex){;}
254                           ClientData clientData = (ClientData) key.attachment();
255                           clientData.header = null;
256                           clientData.body = null;
257                           clientData.hdr = null;
258                           clientData.conectContex = null;
259                           clientData.clientSocketAddress = null;
260                           key.attach(null);
261                           key.cancel();
262                           i.remove();
263                         }
264                     }
265                 }
266                 //gc();
267             }
268         } catch (IOException e) {
269             LOG.error(e);
270             System.exit(1);
271         } finally {
272             if (server != null) {
273                 try {
274                     server.close();
275                 } catch (IOException e) {
276                     e.printStackTrace();
277                 }
278             }
279         }

281     private void processAccept(SelectionKey key) {
282         try {
283 
284             ServerSocketChannel server = (ServerSocketChannel)key.channel();
285             SocketChannel client = server.accept();
286             Socket s = client.socket();
287             // Accept the request
288             this.metrics.listenerStartRequest(System.nanoTime());
289             if(LOG.isDebugEnabled())
290                 LOG.debug("Received an incoming connection from: " + s.getRemoteSocketAddress());
291             client.configureBlocking( false );
292             SelectionKey clientkey = client.register( selector, SelectionKey.OP_READ );
293             clientkey.attach(new ClientData(s.getRemoteSocketAddress()));
294             if(LOG.isDebugEnabled())
295                 LOG.debug(s.getRemoteSocketAddress() + ": " + "Accept processed");
296         } catch (IOException ie) {
297             LOG.error("Cannot Accept connection: " + ie.getMessage());
298         }
299     }

301     private void processRead(SelectionKey key) {
302 
303         key.interestOps( key.interestOps() ^ SelectionKey.OP_READ);
304 
305         SocketChannel client = (SocketChannel) key.channel();
306         Socket s = client.socket();
307         long readLength=0;
308 
309         ClientData clientData = (ClientData) key.attachment();
310         if (false == timeouts.isEmpty() && true == timeouts.containsKey(key)){
311             timeouts.remove(key);
312         }
313 
314         try {
315             while ((readLength = client.read(clientData.buf)) > 0) {
316                 clientData.total_read += readLength;
317                 if(LOG.isDebugEnabled())
318                     LOG.debug(s.getRemoteSocketAddress() + ": " + "Read readLength  " + readLength + "  total " + clientData.total_read);
319             }
320             if (readLength == -1 ){
321                 throw new IOException(s.getRemoteSocketAddress() + ": " + "Connection closed by peer on READ");
322             }
323 
324             if (clientData.total_read < ListenerConstants.HEADER_SIZE){
325                 key.interestOps(key.interestOps() | SelectionKey.OP_READ);
326                 key.attach(clientData);
327                 timeouts.put(key, System.currentTimeMillis());
328                 if(LOG.isDebugEnabled())
329                     LOG.debug(s.getRemoteSocketAddress() + ": " + "Read length less than HEADER size. Added timeout on READ");
330                 return;
331             }
332 
333             if (ListenerConstants.BUFFER_INIT == clientData.buffer_state){
334                 clientData.buffer_state = ListenerConstants.HEADER_PROCESSED;
335                 ByteBuffer hdr = clientData.buf[0];
336                 hdr.flip();
337                 clientData.hdr.extractFromByteArray(hdr);
338                 if (clientData.hdr.signature_ == ListenerConstants.LITTLE_ENDIAN_SIGNATURE){
339                     clientData.switchEndian();
340                     hdr = clientData.buf[0];
341                     clientData.hdr.extractFromByteArray(hdr);
342                 }
343                 if (clientData.hdr.signature_ != ListenerConstants.BIG_ENDIAN_SIGNATURE)
344                     throw new IOException("Wrong signature in read Header : " + clientData.hdr.signature_);
345             }
346             if (clientData.total_read < (clientData.hdr.getTotalLength() + ListenerConstants.HEADER_SIZE)){
347                 key.interestOps(key.interestOps() | SelectionKey.OP_READ);
348                 key.attach(clientData);
349                 timeouts.put(key, System.currentTimeMillis());
350                 if(LOG.isDebugEnabled())
351                     LOG.debug(s.getRemoteSocketAddress() + ": " + "Total length less than in read HEADER. Added timeout on READ");
352                 return;
353             }
354             if (clientData.total_read > (clientData.hdr.getTotalLength() + ListenerConstants.HEADER_SIZE)){
355                 throw new IOException("Wrong total length in read Header : total_read " + clientData.total_read + ", hdr_total_length + hdr_size " + clientData.hdr.getTotalLength() +  + ListenerConstants.HEADER_SIZE);
356             }
357             key.attach(clientData);
358             this.worker.processData(this, key);
359             if(LOG.isDebugEnabled())
360                 LOG.debug(s.getRemoteSocketAddress() + ": " + "Read processed");
361 
362         } catch (IOException ie){
363             LOG.error("IOException: " + s.getRemoteSocketAddress() + ": " + ie.getMessage());
364             try {
365                 client.close();
366             } catch (IOException ex){;}
367             clientData = (ClientData) key.attachment();
368             clientData.header = null;
369             clientData.body = null;
370             clientData.hdr = null;
371             clientData.conectContex = null;
372             clientData.clientSocketAddress = null;
373             key.attach(null);
374             key.cancel();
375             if(metrics != null) metrics.listenerRequestRejected();
376         }
377     }

379     private void processWrite(SelectionKey key) {
380     
381         key.interestOps(key.interestOps() ^ SelectionKey.OP_WRITE);
382 
383         SocketChannel client = (SocketChannel) key.channel();
384         Socket s = client.socket();
385         long writeLength=0;
386 
387         ClientData clientData = (ClientData) key.attachment();
388         if (false == timeouts.isEmpty() && true == timeouts.containsKey(key)){
389             timeouts.remove(key);
390         }
391         try {
392             while ((writeLength = client.write(clientData.buf)) > 0) {
393                 clientData.total_write += writeLength;
394             }
395             if (writeLength == -1 ){
396                 throw new IOException(s.getRemoteSocketAddress() + ": " + "Connection closed by peer on WRITE");
397             }
398 
399             if (clientData.buf[0].hasRemaining() || clientData.buf[1].hasRemaining()){
400                 key.attach(clientData);
401                 key.interestOps(key.interestOps() | SelectionKey.OP_WRITE);
402                 timeouts.put(key, System.currentTimeMillis());
403                 if(LOG.isDebugEnabled())
404                     LOG.debug(s.getRemoteSocketAddress() + ": " + "Write length less than total write length. Added timeout on WRITE");
405             } else {
406                 client.close();
407                 clientData.header = null;
408                 clientData.body = null;
409                 clientData.hdr = null;
410                 clientData.conectContex = null;
411                 clientData.clientSocketAddress = null;
412                 key.attach(null);
413                 key.cancel();
414                 if(metrics != null)metrics.listenerEndRequest(System.nanoTime());
415                 if(LOG.isDebugEnabled())
416                     LOG.debug(s.getRemoteSocketAddress() + ": " + "Write processed");
417             }
418         } catch (IOException ie){
419             LOG.error("IOException: " + s.getRemoteSocketAddress() + ": " + ie.getMessage());
420             try {
421                 client.close();
422             } catch (IOException ex){;}
423             clientData.header = null;
424             clientData.body = null;
425             clientData.hdr = null;
426             clientData.conectContex = null;
427             clientData.clientSocketAddress = null;
428             key.attach(null);
429             key.cancel();
430             if(metrics != null)metrics.listenerRequestRejected();
431         }
