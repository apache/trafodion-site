63  		String user = conf.get("hpt4jdbc.properties.user","");
64  		String password = conf.get("hpt4jdbc.properties.password","");
65  		
66  		ExecutorService tpes = Executors.newCachedThreadPool();
67  		IntegrationTestEnduranceWorker workers[] = new IntegrationTestEnduranceWorker[numWorkers];
68  		Future futures[] = new Future[numWorkers];
69  
70  		for (int i = 0; i < numWorkers; i++) {
71  			workers[i] = new IntegrationTestEnduranceWorker(i,loops,delay,elapsed,catalog,schema,url,user,password);
72  			futures[i]=tpes.submit(workers[i]);
73  		}
74  		
75  		for (int i = 0; i < numWorkers; i++) {
76  			try {
77  				LOG.info("Ending thread: " + (i + 1) + ", average elapsed time: " + futures[i].get());
78  			} catch (Exception e) {
79  				e.printStackTrace();
80  				LOG.info(e.getMessage());
81  			}
82  		}
83  		
84  		LOG.info("All threads ended");
85  	}

87  	private class IntegrationTestEnduranceWorker implements Callable<Integer> {
88  		private int numWorker;
89  		private int loops;
90  		private int delay;
91  		private int elapsed;
92  		private String catalog;
93  		private String schema;
94  		private String url;
95  		private String user;
96  		private String password;
97  		private long startTotal;
98  		private long start;
99  		private long end;
100 		private Long microseconds;
101 		private Integer average = 0;
102 
103 		public IntegrationTestEnduranceWorker(int numWorker,int loops,int delay,int elapsed,String catalog,String schema,String url,String user,String password){
104 			this.numWorker = numWorker;
105 			this.loops = loops;
106 			this.delay = delay;
107 			this.elapsed = elapsed * 1000; //in milliseconds
108 			this.catalog = catalog;
109 			this.schema = schema;
110 			this.url = url;
111 			this.user = user;
112 			this.password = password;
113 		}
114 
115 		@Override
116 		public Integer call() {
117 
118 			LOG.info("Thread number: " + numWorker);
119 			//------------------------------------------------------------------------
120 			// Load JDBC driver
121 			//
122 			try {
123 				Class.forName("org.trafodion.jdbc.t4.T4Driver");
124 			} catch (ClassNotFoundException e) {
125 				fail("Thread number: " + numWorker + ". Could not find the JDBC driver class." + e);
126 				return average;
127 			}
128 
129 			// Create property object to hold username & password
130 			LOG.info("url=" + this.url + ",user=" + this.user + ",password=" + this.password);
131 			Properties myProp = new Properties();
132 			myProp.put("user", this.user);
133 			myProp.put("password", this.password);
134 			Connection conn;
135 			String jdbcUrl = this.url;
136 
137 			startTotal = System.currentTimeMillis();
138 
139 			int i = 1;
140 			while (true) {
141 				start = System.nanoTime();
142 
143 				try {
144 					conn = DriverManager.getConnection(jdbcUrl, myProp);
145 				} catch (SQLException e) {
146 					fail("Thread number/loop: " + numWorker + "/" + i + ". Could not connect to database." + e);
147 					return average;
148 				}
149 
150 				LOG.info("Thread number/loop: " + numWorker + "/" + i + ". Connection open.");
151 
152 				if (delay != 0){
153 					LOG.info("Thread number: " + numWorker + " sleeps: " + delay + " miliseconds");
154 					try {
155 						Thread.sleep(delay);
156 					} catch (Exception e) {	}
157 				}				
158 				
159 				try {
160 					conn.close();
161 				} catch (SQLException e){
162 					LOG.info("Thread number/loop: " + numWorker + "/" + i + ".Close connection error." + e);
163 					return average;
164 				}
165 
166 				end = System.nanoTime();
167 				microseconds = (end - start) / 1000;
168 				if (average == 0)
169 					average = microseconds.intValue();
170 				else
171 					average = (average + microseconds.intValue())/2;
172 
173 				LOG.info("Thread number/loop: " + numWorker + "/" + i + ". Connection closed. Elapsed time (microseconds): " + microseconds);
174 
175 				if (elapsed == 0){
176 					if (i >= loops ) break;
177 				} else {
178 					if (System.currentTimeMillis() >= startTotal + elapsed ) break;
179 				}
180 				i++;
181 			}
182 
183 			return average;
184 		}
185 	}
