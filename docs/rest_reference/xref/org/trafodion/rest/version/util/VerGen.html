114     public static class Version {
115         public int maj;
116         public int min;
117         public int micro;
118         public String qualifier;
119     }
120     
121     public static Version parseVersionString(String input) {
122         Version result = new Version();
123 
124         Pattern p = Pattern.compile("^(\\d+).(\\d+).(\\d+)(-(.+))?$");
125         Matcher m = p.matcher(input);
126 
127         if (!m.matches()) {
128             return null;
129         }
130         result.maj = Integer.parseInt(m.group(1));
131         result.min = Integer.parseInt(m.group(2));
132         result.micro = Integer.parseInt(m.group(3));
133         if (m.groupCount() == 5) {
134             result.qualifier = m.group(5);
135         } else {
136             result.qualifier = null;
137         }
138         return result;
139     }

141     /**
142      * Emits a org.apache.zookeeper.version.Info interface file with version and
143      * revision information constants set to the values passed in as command
144      * line parameters. The file is created in the current directory. <br>
145      * Usage: java org.apache.zookeeper.version.util.VerGen maj.min.micro[-qualifier]
146      * rev buildDate
147      *
148      * @param args
149      *            <ul>
150      *            <li>maj - major version number
151      *            <li>min - minor version number
152      *            <li>micro - minor minor version number
153      *            <li>qualifier - optional qualifier (dash followed by qualifier text)
154      *            <li>rev - current revision number
155      *            <li>buildDate - date the build
156      *            </ul>
157      */
158     public static void main(String[] args) {
159         if (args.length != 3)
160             printUsage();
161         try {
162             Version version = parseVersionString(args[0]);
163             if (version == null) {
164                 System.err.println(
165                         "Invalid version number format, must be \"x.y.z(-.*)?\"");
166                 System.exit(1);
167             }
168             int rev;
169             try {
170                 rev = Integer.parseInt(args[1]);
171             } catch (NumberFormatException e) {
172                 rev = -1;
173             }
174             generateFile(new File("."), version, rev, args[2]);
175         } catch (NumberFormatException e) {
176             System.err.println(
177                     "All version-related parameters must be valid integers!");
178             throw e;
179         }
180     }
