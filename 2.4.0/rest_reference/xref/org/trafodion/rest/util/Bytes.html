107  /**
108    * Read byte-array written with a WritableableUtils.vint prefix.
109    * @param in Input to read from.
110    * @return byte array read off <code>in</code>
111    * @throws IOException e
112    */
113   public static byte [] readByteArray(final DataInput in)
114   throws IOException {
115     int len = WritableUtils.readVInt(in);
116     if (len < 0) {
117       throw new NegativeArraySizeException(Integer.toString(len));
118     }
119     byte [] result = new byte[len];
120     in.readFully(result, 0, len);
121     return result;
122   }

168   /**
169    * Write byte-array from src to tgt with a vint length prefix.
170    * @param tgt target array
171    * @param tgtOffset offset into target array
172    * @param src source array
173    * @param srcOffset source offset
174    * @param srcLength source length
175    * @return New offset in src array.
176    */
177   public static int writeByteArray(final byte [] tgt, final int tgtOffset,
178       final byte [] src, final int srcOffset, final int srcLength) {
179     byte [] vint = vintToBytes(srcLength);
180     System.arraycopy(vint, 0, tgt, tgtOffset, vint.length);
181     int offset = tgtOffset + vint.length;
182     System.arraycopy(src, srcOffset, tgt, offset, srcLength);
183     return offset + srcLength;
184   }

299   /**
300    * Write a printable representation of a byte array. Non-printable
301    * characters are hex escaped in the format \\x%02X, eg:
302    * \x00 \x05 etc
303    *
304    * @param b array to write out
305    * @param off offset to start at
306    * @param len length to write
307    * @return string output
308    */
309   public static String toStringBinary(final byte [] b, int off, int len) {
310     StringBuilder result = new StringBuilder();
311     try {
312       String first = new String(b, off, len, "ISO-8859-1");
313       for (int i = 0; i < first.length() ; ++i ) {
314         int ch = first.charAt(i) & 0xFF;
315         if ( (ch >= '0' && ch <= '9')
316             || (ch >= 'A' && ch <= 'Z')
317             || (ch >= 'a' && ch <= 'z')
318             || " `~!@#$%^&*()-_=+[]{}\\|;:'\",.<>/?".indexOf(ch) >= 0 ) {
319           result.append(first.charAt(i));
320         } else {
321           result.append(String.format("\\x%02X", ch));
322         }
323       }
324     } catch (UnsupportedEncodingException e) {
325       LOG.error("ISO-8859-1 not supported?", e);
326     }
327     return result.toString();
328   }

330   private static boolean isHexDigit(char c) {
331     return
332         (c >= 'A' && c <= 'F') ||
333         (c >= '0' && c <= '9');
334   }

336   /**
337    * Takes a ASCII digit in the range A-F0-9 and returns
338    * the corresponding integer/ordinal value.
339    * @param ch  The hex digit.
340    * @return The converted hex value as a byte.
341    */
342   public static byte toBinaryFromHex(byte ch) {
343     if ( ch >= 'A' && ch <= 'F' )
344       return (byte) ((byte)10 + (byte) (ch - 'A'));
345     // else
346     return (byte) (ch - '0');
347   }

349   public static byte [] toBytesBinary(String in) {
350     // this may be bigger than we need, but let's be safe.
351     byte [] b = new byte[in.length()];
352     int size = 0;
353     for (int i = 0; i < in.length(); ++i) {
354       char ch = in.charAt(i);
355       if (ch == '\\' && in.length() > i+1 && in.charAt(i+1) == 'x') {
356         // ok, take next 2 hex digits.
357         char hd1 = in.charAt(i+2);
358         char hd2 = in.charAt(i+3);
359 
360         // they need to be A-F0-9:
361         if (!isHexDigit(hd1) ||
362             !isHexDigit(hd2)) {
363           // bogus escape code, ignore:
364           continue;
365         }
366         // turn hex ASCII digit -> number
367         byte d = (byte) ((toBinaryFromHex((byte)hd1) << 4) + toBinaryFromHex((byte)hd2));
368 
369         b[size++] = d;
370         i += 3; // skip 3
371       } else {
372         b[size++] = (byte) ch;
373       }
374     }
375     // resize:
376     byte [] b2 = new byte[size];
377     System.arraycopy(b, 0, b2, 0, size);
378     return b2;
379   }

456   /**
457    * Converts a byte array to a long value.
458    *
459    * @param bytes array of bytes
460    * @param offset offset into array
461    * @param length length of data (must be {@link #SIZEOF_LONG})
462    * @return the long value
463    * @throws IllegalArgumentException if length is not {@link #SIZEOF_LONG} or
464    * if there's not enough room in the array at the offset indicated.
465    */
466   public static long toLong(byte[] bytes, int offset, final int length) {
467     if (length != SIZEOF_LONG || offset + length > bytes.length) {
468       throw explainWrongLengthOrOffset(bytes, offset, length, SIZEOF_LONG);
469     }
470     long l = 0;
471     for(int i = offset; i < offset + length; i++) {
472       l <<= 8;
473       l ^= bytes[i] & 0xFF;
474     }
475     return l;
476   }

478   private static IllegalArgumentException
479     explainWrongLengthOrOffset(final byte[] bytes,
480                                final int offset,
481                                final int length,
482                                final int expectedLength) {
483     String reason;
484     if (length != expectedLength) {
485       reason = "Wrong length: " + length + ", expected " + expectedLength;
486     } else {
487      reason = "offset (" + offset + ") + length (" + length + ") exceed the"
488         + " capacity of the array: " + bytes.length;
489     }
490     return new IllegalArgumentException(reason);
491   }

494    * Put a long value out to the specified byte array position.
495    * @param bytes the byte array
496    * @param offset position in the array
497    * @param val long to write out
498    * @return incremented offset
499    * @throws IllegalArgumentException if the byte array given doesn't have
500    * enough room at the offset specified.
501    */
502   public static int putLong(byte[] bytes, int offset, long val) {
503     if (bytes.length - offset < SIZEOF_LONG) {
504       throw new IllegalArgumentException("Not enough room to put a long at"
505           + " offset " + offset + " in a " + bytes.length + " byte array");
506     }
507     for(int i = offset + 7; i > offset; i--) {
508       bytes[i] = (byte) val;
509       val >>>= 8;
510     }
511     bytes[offset] = (byte) val;
512     return offset + SIZEOF_LONG;
513   }

628    * @param bytes byte array

630    * @param length length of int (has to be {@link #SIZEOF_INT})

635   public static int toInt(byte[] bytes, int offset, final int length) {
636     if (length != SIZEOF_INT || offset + length > bytes.length) {
637       throw explainWrongLengthOrOffset(bytes, offset, length, SIZEOF_INT);

639     int n = 0;
640     for(int i = offset; i < (offset + length); i++) {
641       n <<= 8;
642       n ^= bytes[i] & 0xFF;
643     }
644     return n;
645   }

821    * @return Vint as bytes array.
822    */
823   public static byte [] vintToBytes(final long vint) {
824     long i = vint;
825     int size = WritableUtils.getVIntSize(i);
826     byte [] result = new byte[size];
827     int offset = 0;
828     if (i >= -112 && i <= 127) {
829       result[offset] = (byte) i;
830       return result;
831     }
832 
833     int len = -112;
834     if (i < 0) {
835       i ^= -1L; // take one's complement'
836       len = -120;
837     }
838 
839     long tmp = i;
840     while (tmp != 0) {
841       tmp = tmp >> 8;
842       len--;
843     }
844 
845     result[offset++] = (byte) len;
846 
847     len = (len < -120) ? -(len + 120) : -(len + 112);
848 
849     for (int idx = len; idx != 0; idx--) {
850       int shiftbits = (idx - 1) * 8;
851       long mask = 0xFFL << shiftbits;
852       result[offset++] = (byte)((i & mask) >> shiftbits);
853     }
854     return result;
855   }
