85      public void extractFromByteBuffer(ByteBuffer bbBuf) throws UnsupportedEncodingException {
86          try {
87              datasource = ByteBufferUtils.extractString(bbBuf);
88              catalog= ByteBufferUtils.extractString(bbBuf);
89              schema= ByteBufferUtils.extractString(bbBuf);
90              location= ByteBufferUtils.extractString(bbBuf);
91              userRole= ByteBufferUtils.extractString(bbBuf);
92      
93              accessMode=bbBuf.getShort();
94              autoCommit=bbBuf.getShort();
95              queryTimeoutSec=bbBuf.getInt();
96              idleTimeoutSec=bbBuf.getInt();
97              loginTimeoutSec=bbBuf.getInt();
98              txnIsolationLevel=bbBuf.getShort();
99              rowSetSize=bbBuf.getShort();
100     
101             diagnosticFlag=bbBuf.getInt();
102             processId=bbBuf.getInt();
103     
104             computerName=ByteBufferUtils.extractString(bbBuf);
105             windowText=ByteBufferUtils.extractString(bbBuf);
106     
107             ctxACP=bbBuf.getInt();
108             ctxDataLang=bbBuf.getInt();
109             ctxErrorLang=bbBuf.getInt();
110             ctxCtrlInferNXHAR=bbBuf.getShort();
111 
112             cpuToUse=bbBuf.getShort();
113             cpuToUseEnd=bbBuf.getShort();
114             connectOptions=ByteBufferUtils.extractString(bbBuf);
115 
116             clientVersionList.extractFromByteBuffer(bbBuf);
117             dialogueId = bbBuf.getInt();
118             contextOptions1 = ByteBufferUtils.extractUInt(bbBuf);
119             contextOptions2 = ByteBufferUtils.extractUInt(bbBuf);
120             
121             if(ServerConstants.INCONTEXT_OPT1_SESSIONNAME == (contextOptions1 & ServerConstants.INCONTEXT_OPT1_SESSIONNAME))
122                 sessionName=ByteBufferUtils.extractString(bbBuf);
123             
124             if(ServerConstants.INCONTEXT_OPT1_CLIENT_USERNAME == (contextOptions1 & ServerConstants.INCONTEXT_OPT1_CLIENT_USERNAME))
125                 clientUserName=ByteBufferUtils.extractString(bbBuf);
126 
127             debugConnectionContext("extract");
128         } catch(UnsupportedEncodingException ue){
129             LOG.error("ConnectionContext :UnsupportedEncodingException :" + ue);
130             throw new UnsupportedEncodingException(ue.getMessage());
131         }
132     }
