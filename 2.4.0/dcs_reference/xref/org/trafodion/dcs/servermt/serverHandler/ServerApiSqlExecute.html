171     ClientData processApi(ClientData clientData) {
172         init();
173         this.clientData = clientData;
174 //
175 // ==============process input ByteBuffer===========================
176 //
177         ByteBuffer bbHeader = clientData.bbHeader;
178         ByteBuffer bbBody = clientData.bbBody;
179         Header hdr = clientData.hdr;
180 
181         bbHeader.flip();
182         bbBody.flip();
183 
184         try {
185             hdr.extractFromByteArray(bbHeader);
186 
187             dialogueId = bbBody.getInt();
188             holdableCursor = bbBody.getInt();
189             queryTimeout = bbBody.getInt();
190             inpRowCnt = bbBody.getInt();
191             maxRowsetSize = bbBody.getInt();
192             sqlStmtType = bbBody.getInt();
193             stmtHandle = bbBody.getInt();
194             stmtType = bbBody.getInt();
195             sqlString = ByteBufferUtils.extractStringWithCharset(bbBody);
196             cursorName = ByteBufferUtils.extractStringWithCharset(bbBody);
197             stmtLabel = ByteBufferUtils.extractStringWithCharset(bbBody);
198             stmtExplainLabel = ByteBufferUtils.extractString(bbBody);
199 //            txId = ByteBufferUtils.extractByteArray(bbBody);
200             if(LOG.isDebugEnabled()) {
201                 LOG.debug(serverWorkerName + ". dialogueId :" + dialogueId);
202                 LOG.debug(serverWorkerName + ". holdableCursor :" + holdableCursor);
203                 LOG.debug(serverWorkerName + ". queryTimeout :" + queryTimeout);
204                 LOG.debug(serverWorkerName + ". inpRowCnt :" + inpRowCnt);
205                 LOG.debug(serverWorkerName + ". maxRowsetSize :" + maxRowsetSize);
206                 LOG.debug(serverWorkerName + ". sqlStmtType :" + sqlStmtType);
207                 LOG.debug(serverWorkerName + ". stmtHandle :" + stmtHandle);
208                 LOG.debug(serverWorkerName + ". stmtType :" + stmtType);
209                 LOG.debug(serverWorkerName + ". sqlString :" + sqlString);
210                 LOG.debug(serverWorkerName + ". cursorName :" + cursorName);
211                 LOG.debug(serverWorkerName + ". stmtLabel :" + stmtLabel);
212                 LOG.debug(serverWorkerName + ". stmtExplainLabel :" + stmtExplainLabel);
213 //                LOG.debug(serverWorkerName + ". txId :" + txId);
214             }
215             if (dialogueId < 1 ) {
216                 throw new SQLException(serverWorkerName + ". Wrong dialogueId :" + dialogueId);
217             }
218             if (dialogueId != clientData.getDialogueId() ) {
219                 throw new SQLException(serverWorkerName + ". Wrong dialogueId sent by the Client [sent/expected] : [" + dialogueId + "/" + clientData.getDialogueId() + "]");
220             }
221 //=============================================================================
222             trafConn = clientData.getTrafConnection();
223             trafStmt = trafConn.getTrafStatement(stmtLabel, stmtHandle);
224 
225             isResultSet = trafStmt.getIsResultSet();
226             isSpj = trafStmt.getIsSpj();
227             stmt = trafStmt.getStatement();
228             if (isSpj == true) cstmt = (CallableStatement)stmt; else pstmt = (PreparedStatement)stmt;
229 //
230 //=====================Process ServerApiSqlExecute===========================
231 //
232             try {
233                 int addedBatches = 0;
234                 paramCount = trafStmt.getParamCount();
235                 if (paramCount > 0) {
236                     long startOffset;
237                     int rowCount;
238                     int paramRowCount;
239                     int paramMode;
240 
241                     if(LOG.isDebugEnabled())
242                     ByteBufferUtils.toHexString("inValues", bbBody, 10);
243                     paramDescList = trafStmt.getParamDescList();
244                     paramLength = trafStmt.getParamLength();
245                     paramValuesLength = bbBody.getInt();
246                     if(LOG.isDebugEnabled()) {
247                         LOG.debug("paramValuesLength :" + paramValuesLength);
248                         LOG.debug("paramLength :" + paramLength);
249                     }
250                     if (paramValuesLength > 0) {
251                         startOffset = bbBody.position();
252                         rowCount = inpRowCnt == 0 ? 1 : inpRowCnt;
253                         paramRowCount = rowCount;
254                         paramMode = 0;
255                         for (int row = 0; row < rowCount; row++) {
256                             for (int param = 1; param <= paramCount; param++) {
257                                 paramDesc = paramDescList.getDescriptors2()[param - 1];
258                                 paramMode = paramDesc.getParamMode();
259                                 if ( paramMode == java.sql.ParameterMetaData.parameterModeOut) continue;
260                                 bbBody = setData(paramDesc, paramRowCount, param, row, bbBody, startOffset);
261                             }
262                             if (rowCount > 1) {
263                                 pstmt.addBatch(); // no SPJ
264                                 addedBatches++;
265                             }
266                             if(LOG.isDebugEnabled())
267                             LOG.debug("endof row :" + row + " addedBatches :" + addedBatches);
268                         }
269                     }
270                     if(LOG.isDebugEnabled())
271                         LOG.debug("addedBatches :" + addedBatches);
272 
273                     for (int param = 1; param <= paramCount; param++) {
274                         paramDesc = paramDescList.getDescriptors2()[param - 1];
275                         paramMode = paramDesc.getParamMode();
276                         if ( paramMode == java.sql.ParameterMetaData.parameterModeIn) continue;
277                         if(LOG.isDebugEnabled())
278                         LOG.debug("registerOutParameter :" + param);
279                         regParamCount++;
280                         cstmt.registerOutParameter(param, java.sql.Types.VARCHAR);
281                     }
282                 }
283                 int txIdLen = bbBody.limit() - bbBody.position();
284                 if (txIdLen > 0) {
285                     txId = new byte[txIdLen];
286                     bbBody.get(txId);
287                 }
288                 if(LOG.isDebugEnabled())
289                     LOG.debug(serverWorkerName + ". txId :" + Arrays.toString(txId));
290 
291                 if (addedBatches > 1) {
292 
293                     if(LOG.isDebugEnabled())
294                         LOG.debug(serverWorkerName + ". T2 Execute.executeBatch()");
295 
296                     int[] numStatus = pstmt.executeBatch();
297                     rowsAffected = numStatus.length;
298 
299                     if(LOG.isDebugEnabled())
300                         LOG.debug(serverWorkerName + ". T2 Execute.executeBatch rowsAffected :" + rowsAffected);
301                 }
302                 else if (isResultSet) {
303                     if(LOG.isDebugEnabled())
304                         LOG.debug(serverWorkerName + ". T2 Execute.executeQuery()");
305                     rs = pstmt.executeQuery();
306                     if (rs != null) {
307                         outDescList = trafStmt.getOutDescList();
308                         trafStmt.closeAllTResultSets();
309                         trafStmt.addTResultSet(new TrafResultSet(rs, 0, stmtLabel, 0, trafStmt.getOutDescList(),""));
310                     }
311                 } else if (isSpj == true) {
312                     if(LOG.isDebugEnabled())
313                         LOG.debug(serverWorkerName + ". SPJ T2 Execute.execute()");
314                     boolean rsAvailable = cstmt.execute();
315                     if (regParamCount > 0) { //out, inout parameters
316                         rowsAffected = 1;
317                         outValuesLength = (int)paramLength;
318                         outValues = new byte[outValuesLength];
319                         outValues = buildOutValues(paramDescList, outValues, paramCount, cstmt, bbBody.order());
320                     }
321                     if(LOG.isDebugEnabled())
322                         LOG.debug(serverWorkerName + ". SPJ T2 Execute.rsAvailable :" + rsAvailable);
323                     if (rsAvailable == true) {
324                         setMultipleResultSets(cstmt);
325                     }
326                 } else {
327                     resultSetCount = 0;
328                     rowsAffected = pstmt.executeUpdate();
329                     if(LOG.isDebugEnabled())
330                         LOG.debug(serverWorkerName + ". T2 Execute.executeUpdate() rowsAffected :" + rowsAffected);
331                 }
332             } catch (BatchUpdateException bex) {
333                 LOG.error(serverWorkerName + ". Execute.BatchUpdateException " + bex);
334                 errorList = new SQLWarningOrErrorList(bex, bex.getUpdateCounts());
335                 returnCode = errorList.getReturnCode();
336 
337             } catch (SQLException ex) {
338                 LOG.error(serverWorkerName + ". Execute.SQLException " + ex);
339                 errorList = new SQLWarningOrErrorList(ex);
340                 returnCode = errorList.getReturnCode();
341             }
342             if (sqlString.isEmpty() == false)
343                 sqlQueryType = SqlUtils.getSqlStmtType(sqlString);
344             else
345                 sqlQueryType = SqlUtils.getSqlStmtType(sqlStmtType);
346 //
347 //===================calculate length of output ByteBuffer========================
348 //
349             if(LOG.isDebugEnabled())
350                 LOG.debug(serverWorkerName + ". calculate length of output ByteBuffer");
351             bbHeader.clear();
352             bbBody.clear();
353 //
354 // check if ByteBuffer is big enough for output
355 //
356             TrafResultSet trs = null;
357 
358             int dataLength = ServerConstants.INT_FIELD_SIZE;//returnCode
359             if (errorList != null)
360                 dataLength += errorList.lengthOfData();
361             else
362                 dataLength += ServerConstants.INT_FIELD_SIZE;//totalErrorLength = 0
363 
364             if (outDescList != null)
365                 dataLength += outDescList.lengthOfData();
366             else
367                 dataLength += ServerConstants.INT_FIELD_SIZE;//outDescLength = 0
368 
369             dataLength += ServerConstants.INT_FIELD_SIZE;//rowsAffected
370             dataLength += ServerConstants.INT_FIELD_SIZE;//sqlQueryType
371             dataLength += ServerConstants.INT_FIELD_SIZE;//estimatedCost
372             dataLength += ByteBufferUtils.lengthOfByteArray(outValues);//outValues
373 
374             dataLength += ServerConstants.INT_FIELD_SIZE;// SPJ Result Sets
375             if (resultSetCount > 0) {
376                 boolean bNextResultSet = true;
377                 trafStmt.setFirstTResultSet();
378                 while (bNextResultSet) {
379                     trs = trafStmt.getTrafResultSet();
380                     dataLength += trs.lengthOfData();
381                     bNextResultSet = trafStmt.getNextTResultSet();
382                 }
383             }
384             dataLength += ByteBufferUtils.lengthOfString(singleSyntax);
385 
386             int availableBuffer = bbBody.capacity() - bbBody.position();
387             if(LOG.isDebugEnabled())
388                 LOG.debug(serverWorkerName + ". dataLength :" + dataLength + " availableBuffer :" + availableBuffer);
389             if (dataLength > availableBuffer )
390                 bbBody = ByteBufferUtils.increaseCapacity(bbBody, dataLength > ServerConstants.BODY_SIZE ? dataLength : ServerConstants.BODY_SIZE );
391 //
392 //===================== build output ==============================================
393 //
394             if(LOG.isDebugEnabled()) {
395                 LOG.debug(serverWorkerName + ". returnCode :" + returnCode);
396                 LOG.debug(serverWorkerName + ". errorList :" + errorList);
397                 LOG.debug(serverWorkerName + ". outDescList :" + outDescList);
398                 LOG.debug(serverWorkerName + ". rowsAffected :" + rowsAffected);
399                 LOG.debug(serverWorkerName + ". sqlQueryType :" + sqlQueryType);
400                 LOG.debug(serverWorkerName + ". estimatedCost :" + estimatedCost);
401                 LOG.debug(serverWorkerName + ". outValues :" + outValues);
402                 LOG.debug(serverWorkerName + ". SPJ Result Sets :" + resultSetCount);
403                 LOG.debug(serverWorkerName + ". singleSyntax :" + singleSyntax);
404             }
405             bbBody.putInt(returnCode);
406             if (errorList != null)
407                 errorList.insertIntoByteBuffer(bbBody);
408             else
409                 bbBody.putInt(0);
410 
411             if (outDescList != null) {
412                 outDescList.insertIntoByteBuffer(bbBody);
413             } else
414                 bbBody.putInt(0);
415 
416             ByteBufferUtils.insertUInt(rowsAffected, bbBody);
417             bbBody.putInt(sqlQueryType);
418             bbBody.putInt(estimatedCost);
419             ByteBufferUtils.insertByteArray(outValues, bbBody); //outValues
420             bbBody.putInt(resultSetCount);
421 
422             if (resultSetCount > 0) {
423                 boolean bNextResultSet = true;
424                 trafStmt.setFirstTResultSet();
425                 while (bNextResultSet) {
426                     trs = trafStmt.getTrafResultSet();
427                     trs.insertIntoByteBuffer(bbBody);
428                     bNextResultSet = trafStmt.getNextTResultSet();
429                 }
430                 trafStmt.setFirstTResultSet();
431             }
432             ByteBufferUtils.insertString(singleSyntax, bbBody);
433 
434             bbBody.flip();
435 //=========================Update header================================
436             hdr.setTotalLength(bbBody.limit());
437             hdr.insertIntoByteBuffer(bbHeader);
438             bbHeader.flip();
439 
440             clientData.setByteBufferArray(bbHeader, bbBody);
441             clientData.setHdr(hdr);
442             clientData.setRequest(ServerConstants.REQUST_WRITE_READ);
443 
444         } catch (UnsupportedEncodingException ue) {
445             LOG.error(serverWorkerName + ". Execute.UnsupportedEncodingException :" + ue);
446             clientData.setRequestAndDisconnect();
447         } catch (Exception e) {
448             LOG.error(serverWorkerName + ". Execute.Exception :" + e);
449             clientData.setRequestAndDisconnect();
450         }
451         reset();
452         return clientData;
453     }

456     ByteBuffer setData(Descriptor2 paramDesc, int paramRowCount, int param, int row, ByteBuffer bbBody, long startOffset) throws SQLException, ParseException, UnsupportedEncodingException {
457         int paramLength = 0;
458 
459         String tmpStr = "";
460         Short tmps = 0;
461         Integer tmpi = 0;
462         Long tmpl = 0L;
463         Float tmpf = 0f;
464         Double tmpd = 0d;
465         BigInteger tmpbi;
466         BigDecimal tmpbd;
467         boolean isSigne = true;
468 
469         Object retObj;
470         byte[] tbuffer = null;
471         int year, month, day, hour, minute, second;
472         long nanoSeconds;
473         String charSet = "";
474 
475         int precision = paramDesc.getPrecision();
476         int scale = paramDesc.getScale();
477         int datetimeCode = paramDesc.getDatetimeCode();
478         int FSDataType = paramDesc.getFsDataType();
479         int OdbcDataType = paramDesc.getOdbcDataType();
480         int dataCharSet = paramDesc.getSqlCharset();
481         int dataLength = paramDesc.getMaxLen();
482         int dataType = paramDesc.getDataType();
483 
484         // setup the offsets
485         int noNullValue = paramDesc.getNoNullValue();
486         int nullValue = paramDesc.getNullValue();
487         long nullOffset = 0L;
488         long noNullOffset = 0L;
489 
490         if (dataType == ServerConstants.SQLTYPECODE_VARCHAR_WITH_LENGTH) {
491             dataLength += 2;
492 
493             if (dataLength % 2 != 0)
494                 dataLength++;
495         }
496         if(LOG.isDebugEnabled())
497             LOG.debug("nullValue :"+ nullValue);
498         if (nullValue != -1) {
499             nullValue = (nullValue * paramRowCount) + (row * 2);
500             nullOffset = startOffset + nullValue;
501             bbBody.position((int)nullOffset);
502             short isNull = bbBody.getShort();
503             if(LOG.isDebugEnabled())
504                 LOG.debug("isNull :"+ isNull);
505             if ( isNull == -1) {
506                 if (isSpj == true) cstmt.setObject(param,null); else pstmt.setObject(param,null);
507                 return bbBody;
508             }
509         }
510         noNullValue = (noNullValue * paramRowCount) + (row * dataLength);
511         noNullOffset = startOffset + noNullValue;
512         bbBody.position((int)noNullOffset);
513 
514         if(LOG.isDebugEnabled())
515             LOG.debug("[" + param + "] noNullOffset :"+ noNullOffset + " noNullValue :" + noNullValue);
516 
517         if(dataCharSet == SqlUtils.SQLCHARSETCODE_UNICODE)
518             charSet = "UTF-16LE";
519         else
520             charSet = SqlUtils.getCharsetName(dataCharSet);
521 
522         switch (dataType) {
523             case ServerConstants.SQLTYPECODE_CHAR:
524                 if(LOG.isDebugEnabled())
525                     LOG.debug("["+ param + "] dataType :SQLTYPECODE_CHAR");
526                 tbuffer = new byte[dataLength];
527                 bbBody.get(tbuffer, 0, dataLength);
528                 retObj = new String(tbuffer, charSet);
529                 if(LOG.isDebugEnabled())
530                     LOG.debug("["+ param + "] dataType :SQLTYPECODE_CHAR length :" + dataLength + " scale :" + scale + "  retObj :" + retObj);
531                 break;
532             case ServerConstants.SQLTYPECODE_VARCHAR:
533             case ServerConstants.SQLTYPECODE_VARCHAR_LONG:
534                 if(LOG.isDebugEnabled())
535                     ByteBufferUtils.toHexString("VARCHAR/VARCHAR_LONG", bbBody, (dataLength > 100 ? 100 : dataLength));
536                 tbuffer = new byte[dataLength];
537                 bbBody.get(tbuffer, 0, dataLength);
538                 retObj = new String(tbuffer, charSet);
539                 if(LOG.isDebugEnabled()) {
540                     LOG.debug("["+ param + "] dataType :SQLTYPECODE_VARCHAR len :" + dataLength + " scale :" + scale + "  retObj :" + retObj);
541                     LOG.debug(" retObj :" + Arrays.toString(((String)retObj).getBytes()));
542                 }
543                 break;
544             case ServerConstants.SQLTYPECODE_VARCHAR_WITH_LENGTH:
545                 if(LOG.isDebugEnabled())
546                     ByteBufferUtils.toHexString("VARCHAR_WITH_LENGTH", bbBody, (dataLength > 100 ? 100 : dataLength));
547                 boolean shortLength = precision < Math.pow(2, 15);
548                 int dataOffset = (shortLength) ? 2 : 4;
549                 dataLength = (shortLength) ? bbBody.getShort() : bbBody.getInt();
550                 tbuffer = new byte[dataLength];
551                 bbBody.get(tbuffer, 0, dataLength);
552                 retObj = new String(tbuffer, charSet);
553                 if(LOG.isDebugEnabled()) {
554                     LOG.debug("["+ param + "] dataType :SQLTYPECODE_VARCHAR len :" + dataLength + " scale :" + scale + "  retObj :" + retObj);
555                     LOG.debug(" retObj :" + Arrays.toString(((String)retObj).getBytes()));
556                 }
557                 break;
558             case ServerConstants.SQLTYPECODE_INTERVAL:
559                 tbuffer = new byte[dataLength];
560                 bbBody.get(tbuffer);
561                 retObj = new String(tbuffer);
562                 if(LOG.isDebugEnabled())
563                     LOG.debug("["+ param + "] dataType :SQLTYPECODE_INTERVAL dataLength :" + dataLength + " scale :" + scale + "  retObj :" + retObj);
564                 break;
565             case ServerConstants.SQLTYPECODE_DATETIME:
566                 switch (datetimeCode) {
567                     case ServerConstants.SQLDTCODE_DATE:
568                         tbuffer = new byte[dataLength];
569                         bbBody.get(tbuffer);
570                         retObj = java.sql.Date.valueOf((new String(tbuffer)).trim());
571                         if(LOG.isDebugEnabled())
572                             LOG.debug("["+ param + "] dataType :SQLDTCODE_DATE dataLength :" + dataLength + " scale :" + scale + "  retObj :" + retObj);
573                         break;
574                     case ServerConstants.SQLDTCODE_TIMESTAMP:
575                         tbuffer = new byte[dataLength];
576                         bbBody.get(tbuffer);
577                         retObj = Timestamp.valueOf((new String(tbuffer)).trim());
578                         if(LOG.isDebugEnabled())
579                             LOG.debug("["+ param + "] dataType :SQLDTCODE_TIMESTAMP dataLength :" + dataLength + " scale :" + scale + "  retObj :" + retObj);
580                         break;
581                     case ServerConstants.SQLDTCODE_TIME:
582                         if (OdbcDataType == java.sql.Types.OTHER) // For Desc.SQLDTCODE_HOUR_TO_FRACTION
583                         {
584                             tbuffer = new byte[dataLength];
585                             bbBody.get(tbuffer);
586                             retObj = new String((new String(tbuffer)).trim());
587                         } else {
588                             dataLength = ServerConstants.timeLength;
589                             tbuffer = new byte[dataLength];
590                             bbBody.get(tbuffer);
591                             retObj = Time.valueOf((new String(tbuffer)).trim());
592                         }
593                         if(LOG.isDebugEnabled())
594                             LOG.debug("["+ param + "] dataType :SQLDTCODE_TIME dataLength :" + dataLength + " scale :" + scale + "  retObj :" + retObj);
595                         break;
596                     default:
597                         tbuffer = new byte[dataLength];
598                         bbBody.get(tbuffer);
599                         retObj = new String(tbuffer);
600                         if(LOG.isDebugEnabled())
601                             LOG.debug("["+ param + "] dataType :default dataLength :" + dataLength + " scale :" + scale + "  retObj :" + retObj);
602                         break;
603                 } // SQLTYPECODE_DATETIME
604                 break;
605             case ServerConstants.SQLTYPECODE_SMALLINT:
606                 short sValue = bbBody.getShort();
607                 retObj = new Short(sValue);
