97      ClientData processRequest(ClientData clientData, Socket s) { 
98          cancelConnection = false;
99          exception = new GetObjRefException();
100         header = clientData.header;
101         body = clientData.body;
102         hdr = clientData.hdr;
103         clientSocketAddress = clientData.clientSocketAddress;
104 
105         try {
106 // get input
107             header.flip();
108             hdr.extractFromByteArray(header);
109             body.flip();
110             dialogueId = body.getInt();
111             srvrType = body.getInt();
112             srvrObjRef = ByteBufferUtils.extractString(body);
113             stopType = body.getInt();
114             if(LOG.isDebugEnabled()){
115                 LOG.debug(clientSocketAddress + ". dialogueId :" + dialogueId);
116                 LOG.debug(clientSocketAddress + ". srvrType :" + srvrType);
117                 LOG.debug(clientSocketAddress + ". srvrObjRef :" + srvrObjRef);
118                 LOG.debug(clientSocketAddress + ". stopType :" + stopType);
119             }
120             String sPort = srvrObjRef;          //JDBC --- port #
121             if (srvrObjRef.startsWith("TCP:")){ //ODBC --- TCP:<IpAddress>/<portNumber>:ODBC
122                 String[] st = srvrObjRef.split(":");
123                 String ip[] = st[1].split("/");
124                 sPort = ip[1];
125             } 
126             port = Integer.parseInt(sPort);
127 // process request
128             String registeredPath = parentZnode + Constants.DEFAULT_ZOOKEEPER_ZNODE_SERVERS_REGISTERED;
129             String nodeRegisteredPath = "";
130             List<String> servers = null;
131             Stat stat = null;
132             String data = "";
133             boolean found = false;
134             String errorText = "";
135             Integer nodeId = 0;
136             Integer processId = 0;
137             
138             if (false == registeredPath.startsWith("/"))
139                 registeredPath = "/" + registeredPath;
140     
141             zkc.sync(registeredPath,null,null);
142             servers =  zkc.getChildren(registeredPath, null);
143             if( ! servers.isEmpty()) {
144                 for(String server : servers) {
145                     nodeRegisteredPath = registeredPath + "/" + server;
146                     stat = zkc.exists(nodeRegisteredPath,false);
147                     if(stat != null){
148                         data = new String(zkc.getData(nodeRegisteredPath, false, stat));
149                         if (false == data.startsWith("CONNECTED:"))
150                             continue;
151                         else {
152                             String[] stData = data.split(":");
153                             if (dialogueId == Long.parseLong(stData[2]) && port == Integer.parseInt(stData[7])){
154                                 nodeId=Integer.parseInt(stData[3]);
155                                 processId=Integer.parseInt(stData[4]);
156                                 found = true;
157                                 break;
158                             }
159                         }
160                     }
161                 }
162                 if (found == true){
163                     if(LOG.isDebugEnabled())
164                         LOG.debug(clientSocketAddress + ". Server found - dialogueId :" + dialogueId + " port :" + port + " nodeId :" + nodeId + " processId :" + processId);
165                         errorText = cancelQuery(nodeId, processId);
166                 } else {
167                     errorText = "Server not found.";
168                     if(LOG.isDebugEnabled())
169                         LOG.debug(clientSocketAddress + ". Server not found - dialogueId :" + dialogueId + " port :" + port);
170                }
171              }
172 // build output
173             header.clear();
174             body.clear();
175             if (false == errorText.equals("")){
176                 exception.exception_nr = ListenerConstants.DcsMasterParamError_exn;
177                 exception.ErrorText = errorText;
178                 if(LOG.isDebugEnabled())
179                     LOG.debug(clientSocketAddress + ": " + exception.ErrorText);
180             }
181             exception.insertIntoByteBuffer(body);
182             body.flip();
183             hdr.setTotalLength(body.limit());
184             hdr.insertIntoByteBuffer(header);
185             header.flip();
186 
187             clientData.header = header;
188             clientData.body = body;
189             clientData.hdr = hdr;
190         } catch (UnsupportedEncodingException ue){
191             if(LOG.isErrorEnabled())
192                 LOG.error("RequestCancelQuery.UnsupportedEncodingException: " + s.getRemoteSocketAddress() + ": " + ue.getMessage(), ue);
193             cancelConnection = true;
194         } catch (KeeperException ke){
195             if(LOG.isErrorEnabled())
196                 LOG.error("RequestCancelQuery.KeeperException: " + s.getRemoteSocketAddress() + ": " + ke.getMessage(), ke);
197             cancelConnection = true;
198         } catch (InterruptedException ie){
199             if(LOG.isErrorEnabled())
200                 LOG.error("RequestCancelQuery.InterruptedException: " + s.getRemoteSocketAddress() + ": " + ie.getMessage(), ie);
201             cancelConnection = true;
202         } catch (BufferUnderflowException e){
203             if(LOG.isErrorEnabled())
204                 LOG.error("RequestCancelQuery.BufferUnderflowException: " + s.getRemoteSocketAddress() + ": " + e.getMessage(), e);
205             cancelConnection = true;
206         }
207         header = null;
208         body = null;
209         hdr = null;
210         clientSocketAddress = null;
211         exception = null;
212 
213         if (cancelConnection == true)
214             clientData.requestReply = ListenerConstants.REQUST_CLOSE;
215         else
216             clientData.requestReply = ListenerConstants.REQUST_WRITE_EXCEPTION;
217         reset();
218         return clientData;
219     }
220     private static final String queryFormatQid = "select cast(substr(variable_info, position('recentQid:' in variable_info) + 11) as varchar(175) CHARACTER SET UTF8) QUERY_ID  from table (statistics(NULL, 'PROCESS_STATS=%d,CPU=%d'))";
221     private static final String queryFormatCancel = "CONTROL QUERY CANCEL QID \"%s\"";

223     String cancelQuery(int nodeId, int processId){
224         java.sql.Connection conn = null;
225         java.sql.Statement stmt1 = null;
226         java.sql.Statement stmt2 = null;
227         java.sql.ResultSet rs = null; 
228         String query = "";
229         String errorText = "";
230         
231         try {
232             query = String.format(queryFormatQid, processId, nodeId);
233             conn = DriverManager.getConnection(Constants.T2_DRIVER_URL);
234             stmt1 = conn.createStatement();
235             stmt2 = conn.createStatement();
236             rs = stmt1.executeQuery(query);
237             if (rs.next()){
238                 String queryId = rs.getString("QUERY_ID").trim();
239                 if(LOG.isDebugEnabled())
240                     LOG.debug("cancelQuery.queryId :" + queryId);
241                 if(true == queryId.endsWith("_PUBLICATION")){
242                     if(LOG.isDebugEnabled())
243                         LOG.debug("cancelQuery: Publication Query - Cancel Query Request is ignored.");
244                 }
245                 else {
246                     query = String.format(queryFormatCancel, queryId);
247                     if(LOG.isDebugEnabled())
248                         LOG.debug("cancelQuery.query :" + query);
249                     stmt2.execute(query);
250                 }
251             }
252             else {
253                 errorText = "QueryId not found for [" + nodeId + "/" + processId + "]";
254                 if(LOG.isDebugEnabled())
255                     LOG.debug("cancelQuery.errorText :" + errorText);
256             }
257         } catch (SQLException se){
258             SQLException nextException;
259             nextException = se;
260             StringBuilder sb = new StringBuilder();
261             do {
262                 sb.append(nextException.getMessage());
263                 sb.append("\nSQLState   " + nextException.getSQLState());
264                 sb.append("\nError Code " + nextException.getErrorCode());
265             } while ((nextException = nextException.getNextException()) != null);
266             errorText = sb.toString();
267             if(LOG.isDebugEnabled())
268                 LOG.debug("cancelQuery.SQLException :" + errorText);
269         }
270         finally {
271             if (conn != null)
272                 try { conn.close(); } catch(SQLException e){ conn = null;}
273         }
274         return errorText;
275     }
