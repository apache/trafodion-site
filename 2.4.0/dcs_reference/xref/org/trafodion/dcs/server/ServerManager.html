92      class RegisteredWatcher implements Watcher {
93          CountDownLatch startSignal;
94  
95          public RegisteredWatcher(CountDownLatch startSignal) {
96              this.startSignal = startSignal;
97          }
98  
99          public void process(WatchedEvent event) {
100             if (event.getType() == Event.EventType.NodeDeleted) {
101                 String znodePath = event.getPath();
102                 if (LOG.isDebugEnabled()) {
103                     LOG.debug("Registered znode deleted [" + znodePath + "]");
104                 }
105                 try {
106                     startSignal.countDown();
107                 } catch (Exception e) {
108                     e.printStackTrace();
109                     LOG.error(e);
110                 }
111             }
112         }
113     }

115     class ServerMonitor {
116         ScriptContext scriptContext = new ScriptContext();
117         int childInstance;
118         String registeredPath;
119         Stat stat = null;
120         boolean isRunning = false;
121         String nid;
122         String pid;
123 
124         public ServerMonitor(int childInstance, String registeredPath) {
125             this.childInstance = childInstance;
126             this.registeredPath = registeredPath;
127         }
128 
129         public boolean monitor() {
130             try {
131                 if (LOG.isDebugEnabled()) {
132                     LOG.debug("registered path [" + registeredPath + "]");
133                 }
134                 stat = zkc.exists(registeredPath, false);
135                 if (stat != null) { // User program znode found in
136                                     // /registered...check pid
137                     isRunning = isPidRunning();
138                     if (LOG.isDebugEnabled()) {
139                         LOG.debug("[" + (Integer.parseInt(nid) + 1) + ":" + childInstance + "]." + pid + ".isRunning ["
140                                 + isRunning + "]");
141                     }
142                 }
143             } catch (Exception e) {
144                 LOG.warn(e.getMessage(), e);
145                 isRunning = false;
146                 return isRunning;
147             }
148 
149             return isRunning;
150         }
151 
152         private boolean isPidRunning() throws Exception {
153             String data = Bytes.toString(zkc.getData(registeredPath, false,
154                     stat));
155             Scanner scn = new Scanner(data);
156             scn.useDelimiter(":");
157             scn.next();// state
158             scn.next();// timestamp
159             scn.next();// dialogue Id
160             nid = scn.next();// nid
161             pid = scn.next();// pid
162             scn.close();
163             scriptContext.setHostName(hostName);
164             scriptContext.setScriptName(Constants.SYS_SHELL_SCRIPT_NAME);
165             scriptContext.setCommand("ps -p " + pid);
166             ScriptManager.getInstance().runScript(scriptContext);
167             return scriptContext.getExitCode() != 0 ? false : true;
168         }
169     }

228         public void exec() throws Exception {
229             cleanupZk();
230             LOG.info("Instance : [" + childInstance + "], User program exec [" + scriptContext.getCommand() + "]");
231             ScriptManager.getInstance().runScript(scriptContext);// This will
232                                                                  // block while
233                                                                  // user prog is
234                                                                  // running
235             LOG.info("Instance : [" + childInstance + "], User program exit [" + scriptContext.getExitCode() + "]");
236             StringBuilder sb = new StringBuilder();
237             sb.append("exit code [" + scriptContext.getExitCode() + "]");
238             if (!scriptContext.getStdOut().toString().isEmpty())
239                 sb.append(", stdout [" + scriptContext.getStdOut().toString()
240                         + "]");
241             if (!scriptContext.getStdErr().toString().isEmpty())
242                 sb.append(", stderr [" + scriptContext.getStdErr().toString()
243                         + "]");
244             LOG.info(sb.toString());
245 
246             switch (scriptContext.getExitCode()) {
247             case 3:
248                 LOG.error("Trafodion is not running");
249                 break;
250             case 127:
251                 LOG.error("Cannot find user program executable");
252                 break;
253             default:
254             }
255         }

257         private void cleanupZk() {
258             try {
259                 Stat stat = zkc.exists(registeredPath, false);
260                 if (stat != null)
261                     zkc.delete(registeredPath, -1);
262             } catch (KeeperException | InterruptedException e) {
263                 e.printStackTrace();
264                 LOG.error(e.getMessage(), e);
265             }
266         }

269     class ServerHandler implements Callable<Integer> {
270         ServerMonitor serverMonitor;
271         ServerRunner serverRunner;
272         int childInstance;
273         String registeredPath;
274         CountDownLatch startSignal = new CountDownLatch(1);
275         RetryCounter retryCounter;
276 
277         public ServerHandler(Configuration conf ,int childInstance) {
278             int maxRestartAttempts = conf.getInt(Constants.DCS_SERVER_STARTUP_MXOSRVR_USER_PROGRAM_RESTART_HANDLER_ATTEMPTS,
279                     Constants.DEFAULT_DCS_SERVER_STARTUP_MXOSRVR_USER_PROGRAM_RESTART_HANDLER_ATTEMPTS);
280             int retryTimeoutMinutes = conf.getInt(
281                     Constants.DCS_SERVER_STARTUP_MXOSRVR_USER_PROGRAM_RESTART_HANDLER_RETRY_TIMEOUT_MINUTES,
282                     Constants.DEFAULT_DCS_SERVER_STARTUP_MXOSRVR_USER_PROGRAM_RESTART_HANDLER_RETRY_TIMEOUT_MINUTES);
283             this.childInstance = childInstance;
284             this.registeredPath = parentZnode
285                     + Constants.DEFAULT_ZOOKEEPER_ZNODE_SERVERS_REGISTERED
286                     + "/" + hostName + ":" + instance + ":" + childInstance;
287             retryCounter = RetryCounterFactory.create(maxRestartAttempts, retryTimeoutMinutes, TimeUnit.MINUTES);
288             serverMonitor = new ServerMonitor(childInstance, registeredPath);
289             serverRunner = new ServerRunner(childInstance, registeredPath);
290         }
291 
292         @Override
293         public Integer call() throws Exception {
294             Integer result = new Integer(childInstance);
295 
296             if (serverMonitor.monitor()) {
297                 LOG.info("Server handler [" + instance + ":" + childInstance
298                         + "] is running");
299                 zkc.exists(registeredPath, new RegisteredWatcher(startSignal));
300                 if (LOG.isDebugEnabled()) {
301                     LOG.debug("Waiting for start signal");
302                 }
303                 startSignal.await();
304                 serverRunner.exec();
305             } else {
306                 LOG.info("Server handler [" + instance + ":" + childInstance
307                         + "] is not running");
308                 serverRunner.exec();
309             }
310 
311             return result;
312         }
313     }

384     private boolean isTrafodionRunning(String nid) {
385 
386         // Check if the given Node is up and running
387         // return true else return false.
388         // Invoke sqcheck -n <nid> to check Node status. 
389         // 
390         //   sqcheck returns:
391         //    -1 - Not up ($?=255) or node down
392         //     0 - Fully up and operational or node up
393         //     1 - Partially up and operational
394         //     2 - Partially up and NOT operational
395  
396         ScriptContext scriptContext = new ScriptContext();
397         scriptContext.setHostName(hostName);
398         scriptContext.setScriptName(Constants.SYS_SHELL_SCRIPT_NAME);
399         String command;
400         if (nid != null)
401            command = ("sqcheck -n " + nid);
402         else
403            command = ("sqcheck");
404         scriptContext.setCommand(command);
405         ScriptManager.getInstance().runScript(scriptContext);// This will
406                                                              // block while
407                                                              // script is
408                                                              // running
409         int exitCode = scriptContext.getExitCode();
410         return (exitCode == 0 || exitCode == 1) ? true : false;
411     }

413     @Override
414     public Boolean call() throws Exception {
415 
416         ExecutorService executorService = Executors
417                 .newFixedThreadPool(childServers);
418         CompletionService<Integer> completionService = new ExecutorCompletionService<Integer>(
419                 executorService);
420 
421         try {
422             getMaster();
423             featureCheck();
424             registerInRunning(instance);
425             RetryCounter retryCounter = RetryCounterFactory.create(maxRestartAttempts, retryIntervalMillis);
426             while (!isTrafodionRunning(null)) {
427                if (!retryCounter.shouldRetry()) {
428                   throw new IOException("Trafodion is not running");
429                } else {
430                   retryCounter.sleepUntilNextRetry();
431                   retryCounter.useRetry();
432                }
433             }
434 
435             // When started from bin/dcs-start.sh script childServers will
436             // contain the
437             // count passed in from the servers.sh script. However when
438             // DcsServer is
439             // killed or dies for any reason DcsMaster restarts it using
440             // /bin/dcs-daemon script
441             // which DOES NOT set childServers count.
442             for (int childInstance = 1; childInstance <= childServers; childInstance++) {
443                 serverHandlers[childInstance-1] = new ServerHandler(conf, childInstance);
444                 completionService.submit(serverHandlers[childInstance-1]);
445                 if (LOG.isDebugEnabled()) {
446                     LOG.debug("Started server handler [" + instance + ":"
447                         + childInstance + "]");
448                 }
449             }
450 
451             while (true) {
452                 if (LOG.isDebugEnabled()) {
453                     LOG.debug("Waiting for any server handler to finish");
454                 }
455                 Future<Integer> f = completionService.take();// blocks waiting
456                                                              // for any
457                                                              // ServerHandler to
458                                                              // finish
459                 if (f != null) {
460                     Integer result = f.get();
461                     LOG.info("Server handler [" + instance + ":" + result + "] exit");
462 
463                     retryCounter = RetryCounterFactory.create(maxRestartAttempts, retryIntervalMillis);
464                     int childInstance = result.intValue();
465                     ServerHandler previousServerHandler = serverHandlers[childInstance - 1];
466                     String nid = serverHandlers[childInstance - 1].serverMonitor.nid;
467                     while (!isTrafodionRunning(nid)) {
468                         if (!retryCounter.shouldRetry()) {
469                             throw new IOException("Node " + nid + " is not Up");
470                         } else {
471                             retryCounter.sleepUntilNextRetry();
472                             retryCounter.useRetry();
473                         }
474                     }
475                     // get the node id
476                     if (previousServerHandler.retryCounter.shouldRetryInnerMinutes()) {
477                         serverHandlers[childInstance - 1] = previousServerHandler;
478                         completionService.submit(serverHandlers[childInstance - 1]);
479                     } else {
480                         serverHandlers[childInstance - 1] = null;
481                     }
482                 }
483             }
484         } catch (Exception e) {
485             e.printStackTrace();
486             LOG.error(e);
487             if (executorService != null)
488                 executorService.shutdown();
489             throw e;
490         }
491         /*
492          * ExecutorService pool = Executors.newSingleThreadExecutor();
493          * 
494          * try { getMaster(); registerInRunning(); featureCheck();
495          * 
496          * Callable<Boolean> serverMonitor = new ServerMonitor();
497          * Callable<ScriptContext> serverRunner = new ServerRunner();
498          * 
499          * long timeoutMillis=5000;
500          * 
501          * while(true) { Future<Boolean> monitor = pool.submit(serverMonitor);
502          * if(false == monitor.get().booleanValue()) { //blocking call
503          * LOG.info("User program is not running"); Future<ScriptContext> runner
504          * = pool.submit(serverRunner); ScriptContext scriptContext =
505          * runner.get();//blocking call
506          * 
507          * StringBuilder sb = new StringBuilder(); sb.append("exit code [" +
508          * scriptContext.getExitCode() + "]"); if(!
509          * scriptContext.getStdOut().toString().isEmpty())
510          * sb.append(", stdout [" + scriptContext.getStdOut().toString() + "]");
511          * if(! scriptContext.getStdErr().toString().isEmpty())
512          * sb.append(", stderr [" + scriptContext.getStdErr().toString() + "]");
513          * LOG.info(sb.toString());
514          * 
515          * switch(scriptContext.getExitCode()) { case 3:
516          * LOG.error("Trafodion is not running"); timeoutMillis=60000; break;
517          * case 127: LOG.error("Cannot find user program executable");
518          * timeoutMillis=60000; break; default: timeoutMillis=5000; }
519          * 
520          * } else { timeoutMillis=5000; }
521          * 
522          * try { Thread.sleep(timeoutMillis); } catch (InterruptedException e) {
523          * } }
524          * 
525          * } catch (Exception e) { e.printStackTrace(); LOG.error(e);
526          * pool.shutdown(); throw e; }
527          */
528     }

530     private void featureCheck() {
531         final String msg1 = "Property "
532                 + Constants.DCS_SERVER_USER_PROGRAM
533                 + " is false. "
534                 + "Please add to your dcs-site.xml file and set <value>false</value> to <value>true</value>.";
535         final String msg2 = "Environment variable $TRAF_HOME is not set.";
536 
537         boolean ready = false;
538         while (!ready) {
539             userProgEnabled = conf.getBoolean(
540                     Constants.DCS_SERVER_USER_PROGRAM,
541                     Constants.DEFAULT_DCS_SERVER_USER_PROGRAM);
542             userProgramHome = System.getProperty("dcs.user.program.home");
543             userProgramCommand = conf.get(
544                     Constants.DCS_SERVER_USER_PROGRAM_COMMAND,
545                     Constants.DEFAULT_DCS_SERVER_USER_PROGRAM_COMMAND);
546 
547             if (userProgEnabled == true && userProgramHome.isEmpty() == false
548                     && userProgramCommand.isEmpty() == false) {
549                 ready = true;
550                 continue;
551             }
552 
553             if (userProgEnabled == false)
554                 LOG.error(msg1);
555             if (userProgramHome.isEmpty())
556                 LOG.error(msg2);
557 
558             try {
559                 Thread.sleep(60000);
560             } catch (InterruptedException e) {
561             }
562         }
563 
564         LOG.info("User program enabled");
565     }

567     private void getMaster() {
568         boolean found = false;
569 
570         while (!found) {
571             LOG.info("Checking DcsMaster znode [" + parentZnode
572                     + Constants.DEFAULT_ZOOKEEPER_ZNODE_MASTER + "]");
573             try {
574                 Stat stat = zkc.exists(parentZnode
575                         + Constants.DEFAULT_ZOOKEEPER_ZNODE_MASTER, false);
576                 if (stat != null) {
577                     List<String> nodes = zkc.getChildren(parentZnode
578                             + Constants.DEFAULT_ZOOKEEPER_ZNODE_MASTER, null);
579                     if (!nodes.isEmpty()) {
580                         StringTokenizer st = new StringTokenizer(nodes.get(0),
581                                 ":");
582                         while (st.hasMoreTokens()) {
583                             masterHostName = st.nextToken();
584                             port = Integer.parseInt(st.nextToken());
585                             portRange = Integer.parseInt(st.nextToken());
586                             masterStartTime = Long.parseLong(st.nextToken());
587                         }
588                         found = true;
589                         LOG.info("DcsMaster znode [" + parentZnode
590                                 + Constants.DEFAULT_ZOOKEEPER_ZNODE_MASTER
591                                 + "] found");
592                     }
593                 } else {
